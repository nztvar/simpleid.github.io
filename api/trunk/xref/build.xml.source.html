 119                      <regexp pattern="md5sum=&quot;[a-z0-9]{32}&quot; " replace="" />
 120                      <!-- remove baseinstalldir for files -->
 121                      <regexp pattern="&lt;file baseinstalldir=&quot;/&quot;" replace="&lt;file"/>
 122                      <!-- install-as for different directories -->
 123                      <regexp pattern="(&lt;file name=&quot;/www/(.+?)&quot;)" replace="\1 install-as=&quot;simpleid/\2&quot;"/>
