  38   *  Stores data into the cache.

  46  function cache_set($type, $key, $data, $time = NULL) {
  47      $filename = _cache_get_name($type, $key);
  48      if (!file_exists(dirname($filename))) mkdir(dirname($filename), 0775, true);
  49      $file = fopen($filename, 'w');
  50      fwrite($file, serialize($data));
  51      fclose($file);
  52      
  53      if ($time != NULL) {
  54          touch($filename, $time);
  55      }

  58  /**
  59   * Obtains data from the cache.
  60   *
  61   * @param string $type the type of data in the cache
  62   * @param string $key an identifier
  63   * @return mixed the data associated with the type and key, or NULL if the cache
  64   * does not contain the requested data.

  66  function cache_get($type, $key) {
  67      $filename = _cache_get_name($type, $key);
  68      
  69      if (!file_exists($filename)) return NULL;
  70      
  71      return unserialize(file_get_contents($filename));
  72  }

  74  /**
  75   * Obtains all data of a particular type from the cache.
  76   *
  77   * @param string $type the type of data in the cache
  78   * @return mixed an array of data associated with the type, or NULL if the cache
  79   * does not contain the requested data.
  80   */
  81  function cache_get_all($type) {
  82      $r = array();
  83      
  84      $dir = opendir(CACHE_DIR . '/' . $type);
  85      
  86      while (($file = readdir($dir)) !== false) {
  87          $filename = CACHE_DIR . '/' . $type . '/' . $file;
  88          
  89          if (filetype($filename) != "file") continue;
  90          
  91          $r[] = unserialize(file_get_contents($filename));
  92      }
  93      
  94      closedir($dir);
  95      
  96      return $r;
  97  }

 110  /**
 111   * Garbage collects data stored the cache.  Data is deleted if it was stored
 112   * for longer than the specified expiry.
 113   *
 114   * This function is deprecated, use {@link cache_expire()}.
 115   *
 116   * @param int $expiry the expiry time, in seconds, after which data will be deleted
 117   * @param string $type the type of data in the cache
 118   * @deprecated
 119   */
 120  function cache_gc($expiry, $type = NULL) {
 121      if ($type == NULL) {
 122          $dir = opendir(CACHE_DIR);
 123  
 124          while (($file = readdir($dir)) !== false) {
 125              $filename = CACHE_DIR . '/' . $file;
 126              if (in_array(filetype($filename), array('dir', 'link')))
 127                  cache_gc($expiry, $file);
 128          }
 129      } else {
 130          $dir = opendir(CACHE_DIR . '/' . $type);
 131          while (($file = readdir($dir)) !== false) {
 132              $filename = CACHE_DIR . '/' . $type . '/' . $file;
 133          
 134              if ((filetype($filename) == "file") && (filectime($filename) < time() - $expiry))
 135                  unlink($filename);
 136          }
 137      }
 138      
 139      closedir($dir);
 140  }

 142  /**
 143   * Garbage collects data stored the cache.  Data is deleted if it was stored
 144   * for longer than the specified expiry.
 145   *
 146   * The parameter to this function takes either an integer or an array.  If the
 147   * parameter is an integer, everything in the cache older than the specified
 148   * time (in seconds) will be deleted.  If the parameter is an array, 
 149   * cache items of the type specified in the key to the array, older than the
 150   * corresponding value will be deleted.
 151   *
 152   * This function is deprecated, use {@link cache_expire()}.
 153   *
 154   * @param int|array $params the expiry time, in seconds, after which data will be deleted,
 155   * or an array specifiying the expiry time for each type
 156   */
 157  function cache_expire($params) {
 158      $dir = opendir(CACHE_DIR);
 159      
 160      while (($file = readdir($dir)) !== false) {
 161          $expiry = NULL;
 162          $filename = CACHE_DIR . '/' . $file;
 163          
 164          if (is_int($params)) {
 165              $expiry = $params;
 166          } elseif (is_array($params)) {
 167              foreach ($params as $type => $param) {
 168                  if (strpos($file, $type . '-') === 0) {
 169                      $expiry = $param;
 170                      break;
 171                  }
 172              }
 173          }
 174          
 175          if (!is_null($expiry) && (filetype($filename) == "file") && (filectime($filename) < time() - $expiry)) {
 176              unlink($filename);
 177          }
 178      }
 179      
 180      closedir($dir);
 181  }

 183  /**
 184   * Returns the time remaining, in seconds, before the data associated with the
 185   * type and key become subject to garbage collection by {@link cache_gc()}.
 186   *
 187   * @param string $type the type of data in the cache
 188   * @param string $key an identifier
 189   * @param int $expiry the expiry time, in seconds, which would be passed onto the
 190   * {@link cache_gc()} function
 191   * @return int the time remaining before expiry, rounded downwards,
 192   * or zero if the cache does not contain the requested data
 193   * @since 0.8
 194   */
 195  function cache_ttl($type, $key, $expiry) {
 196      $filename = _cache_get_name($type, $key);
 197      
 198      if (!file_exists($filename)) return 0;
 199      
 200      return filectime($filename) - (time() - $expiry) - 1;
 201  }

 203  /**
 204   * Returns the name of the cache data file, given a type and an identifier.
 205   *
 206   * @param string $type the type of data in the cache
 207   * @param string $key an identifier
 208   * @return string a file name
 209   */
 210  function _cache_get_name($type, $key) {
 211      return CACHE_DIR . '/' . $type . '/' . md5($key) . '.cache';
