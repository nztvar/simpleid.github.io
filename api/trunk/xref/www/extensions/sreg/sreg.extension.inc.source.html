  44  function sreg_response($assertion, $request) {
  45      global $user;
  46      global $version;
  47      
  48      // We only deal with positive assertions
  49      if (!$assertion) return array();
  50      
  51      // We only respond if the extension is requested
  52      if (!openid_extension_requested('http://openid.net/extensions/sreg/1.1', $request)) return array();
  53      
  54      $request = openid_extension_filter_request('http://openid.net/extensions/sreg/1.1', $request);
  55      $required = (isset($request['required'])) ? explode(',', $request['required']) : array();
  56      $optional = (isset($request['optional'])) ? explode(',', $request['optional']) : array();
  57      $fields = array_merge($required, $optional);
  58      $alias = openid_extension_alias('http://openid.net/extensions/sreg/1.1');
  59      $response = array();
  60      
  61      if ($version == OPENID_VERSION_2) $response['openid.ns.' . $alias] = 'http://openid.net/extensions/sreg/1.1';
  62      
  63      foreach ($fields as $field) {
  64          if (isset($user['sreg'][$field])) {
  65              $response['openid.' . $alias . '.' .  $field] = $user['sreg'][$field];
  66          }
  67      }
  68      
  69      return $response;
  70  }

  77  function sreg_signed_fields($response) {
  78      // We only respond if the extension is requested
  79      if (!openid_extension_requested('http://openid.net/extensions/sreg/1.1', $response)) return array();
  80      
  81      $fields = array_keys(openid_extension_filter_request('http://openid.net/extensions/sreg/1.1', $response));
  82      $alias = openid_extension_alias('http://openid.net/extensions/sreg/1.1');
  83      $signed_fields = array();
  84  
  85      if (isset($response['openid.ns.' . $alias])) $signed_fields[] = 'ns.' . $alias;
  86      foreach ($fields as $field) {
  87          if (isset($response['openid.' . $alias . '.' . $field])) $signed_fields[] = $alias . '.' . $field;
  88      }
  89      
  90      return $signed_fields;
  91  }

  96  function sreg_consent_form($request, $response, $rp) {
  97      global $user;
  98      
  99      // We only respond if the extension is requested
 100      if (!openid_extension_requested('http://openid.net/extensions/sreg/1.1', $request)) return '';
 101      
 102      $request = openid_extension_filter_request('http://openid.net/extensions/sreg/1.1', $request);
 103      $required = (isset($request['required'])) ? explode(',', $request['required']) : array();
 104      $optional = (isset($request['optional'])) ? explode(',', $request['optional']) : array();
 105      $keys = array_merge($required, $optional);
 106      
 107      if ((count($request)) && isset($user['sreg'])) {
 108          $xtpl2 = new XTemplate('extensions/sreg/sreg.xtpl');
 109          
 110          $xtpl2->assign('alias', openid_extension_alias('http://openid.net/extensions/sreg/1.1'));
 111          
 112          if (isset($request['policy_url'])) {            
 113              $xtpl2->assign('policy', 'You can view the site\'s policy in relation to the use of this information at this URL: <a href="' . htmlspecialchars($request['policy_url'], ENT_QUOTES, 'UTF-8') . '">' . htmlspecialchars($request['policy_url'], ENT_QUOTES, 'UTF-8') . '</a>.');            
 114          }
 115          
 116          foreach ($keys as $key) {
 117              if (isset($user['sreg'][$key])) {
 118                  $xtpl2->assign('name', htmlspecialchars($key, ENT_QUOTES, 'UTF-8'));
 119                  $xtpl2->assign('value', htmlspecialchars($user['sreg'][$key], ENT_QUOTES, 'UTF-8'));
 120                  $xtpl2->parse('form.sreg');
 121              }
 122          }
 123          
 124          $xtpl2->parse('form');
 125          return $xtpl2->text('form');
 126      }
 127  }

 132  function sreg_page_profile() {
 133      global $user;
 134      $xtpl2 = new XTemplate('extensions/sreg/sreg.xtpl');
 135      
 136      if (isset($user['sreg'])) {

 138              $xtpl2->assign('name', htmlspecialchars($name, ENT_QUOTES, 'UTF-8'));
 139              $xtpl2->assign('value', htmlspecialchars($value, ENT_QUOTES, 'UTF-8'));
 140              $xtpl2->parse('user_page.sreg');
 141          }
 142      }    
 143      
 144      $xtpl2->parse('user_page');
 145      
 146      return array(array(
 147          'id' => 'sreg',
 148          'title' => 'Simple Registration Extension',
 149          'content' => $xtpl2->text('user_page')
