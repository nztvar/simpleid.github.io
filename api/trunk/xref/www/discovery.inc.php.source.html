  62  /**
  63   * Obtains the services for particular identifier.
  64   *
  65   * This function attempts to discover and obtain the XRDS document associated
  66   * with the identifier, parses the XRDS document and returns an array of
  67   * services.
  68   *
  69   * If an XRDS document is not found, and $openid is set to true, this function
  70   * will also attempt to discover OpenID services by looking for link elements
  71   * with rel of openid.server or openid2.provider in the discovered HTML document.
  72   *
  73   * @param string $identifier the identifier
  74   * @param bool $openid if true, performs additional discovery of OpenID services
  75   * by looking for link elements within the discovered document
  76   * @return array an array of discovered services, or an empty array if no services
  77   * are found
  78   */
  79  function discovery_xrds_discover($identifier, $openid = FALSE) {
  80      $identifier = discovery_xrds_normalize($identifier);
  81      $url = discovery_xrds_url($identifier);
  82      
  83      $xrds = discovery_xrds_get($url);
  84  
  85      if ($xrds) {
  86          return discovery_xrds_parse($xrds);
  87      } else {
  88          if ($openid) return discovery_html_get_services($url);
  89          return array();
  90      }

 129  /**
 130   * Obtains a XRDS document at a particular URL.  Performs Yadis discovery if
 131   * the URL does not produce a XRDS document.
 132   *
 133   * @param string $url the URL
 134   * @param bool $check whether to check the content type of the response is
 135   * application/xrds+xml
 136   * @param int $retries the number of tries to make
 137   * @return string the contents of the XRDS document
 138   */
 139  function discovery_xrds_get($url, $check = TRUE, $retries = 5) {
 140      if ($retries == 0) return NULL;
 141      
 142      $response = http_make_request($url, array('Accept' => 'application/xrds+xml'));
 143  
 144      if (isset($response['http-error'])) return NULL;
 145      if (($response['content-type'] == 'application/xrds+xml') || ($check == FALSE)) {
 146          return $response['data'];
 147      } elseif (isset($response['headers']['x-xrds-location'])) {
 148          return discovery_xrds_get($response['headers']['x-xrds-location'], false, $retries - 1);
 149      } else {
 150          $location = _discovery_meta_httpequiv('X-XRDS-Location', $response['data']);
 151          if ($location) {
 152              return discovery_xrds_get($location, false, $retries - 1);
 153          }
 154          return NULL;
 155      }
 156  }

 158  /**
 159   * Normalises an identifier for discovery.
 160   *
 161   * If the identifier begins with xri://, acct: or mailto:, this is stripped out.  If the identifier
 162   * does not begin with a valid URI scheme, http:// is assumed and added to the
 163   * identifier.
 164   *
 165   * @param string $identifier the identifier to normalise
 166   * @return string the normalised identifier
 167   */
 168  function discovery_xrds_normalize($identifier) {
 169      $normalized = $identifier;
 170      
 171      if (discovery_is_xri($identifier)) {
 172          if (stristr($identifier, 'xri://') !== false) $normalized = substr($identifier, 6);
 173      } elseif (discovery_is_email($identifier)) {
 174          if (stristr($identifier, 'acct:') !== false) $normalized = substr($identifier, 5);
 175          if (stristr($identifier, 'mailto:') !== false) $normalized = substr($identifier, 7);
 176      } else {
 177          if (stristr($identifier, '://') === false) $normalized = 'http://'. $identifier;
 178          if (substr_count($normalized, '/') < 3) $normalized .= '/';
 179      }
 180      
 181      return $normalized;
 182  }

 203  /**
 204   * Determines whether an identifier is an XRI.
 205   *
 206   * XRI identifiers either start with xri:// or with @, =, +, $ or !.
 207   *
 208   * @param string $identifier the parameter to test
 209   * @return bool true if the identifier is an XRI
 210   */
 211  function discovery_is_xri($identifier) {
 212      $firstchar = substr($identifier, 0, 1);
 213      if ($firstchar == "@" || $firstchar == "=" || $firstchar == "+" || $firstchar == "\$" || $firstchar == "!") return true;
 214      if (stristr($identifier, 'xri://') !== FALSE) return true;
 215      return false;
 216  }

 218  /**
 219   * Determines whether an identifier is an e-mail address.
 220   *
 221   * An identifier is an e-mail address if it:
 222   *
 223   * - has a single @ character
 224   * - does not have a slash character
 225   *
 226   * @param string $identifier the parameter to test
 227   * @return bool true if the identifier is an e-mail address
 228   */
 229  function discovery_is_email($identifier) {
 230      // If it begins with acct: or mailto:, strip it out
 231      if (stristr($identifier, 'acct:') !== false) $identifier = substr($identifier, 5);
 232      if (stristr($identifier, 'mailto:') !== false) $identifier = substr($identifier, 7);
 233      
 234      // If it contains a slash, it is not an e-mail address
 235      if (strpos($identifier, "/") !== false) return false;
 236      
 237      $at = strpos($identifier, "@");
 238      
 239      // If it does not contain a @, it is not an e-mail address
 240      if ($at === false) return false;
 241      
 242      // If it contains more than one @, it is not an e-mail
 243      if (strrpos($identifier, "@") != $at) return false;
 244      
 245      return true;
 246  }

 249  /**
 250   * Callback function to sort service and URI elements based on priorities
 251   * specified in the XRDS document.
 252   *
 253   * The XRDS specification allows multiple instances of certain elements, such
 254   * as Service and URI.  The specification allows an attribute called priority
 255   * so that the document creator can specify the order the elements should be used.
 256   *
 257   * @param array $a
 258   * @param array $b
 259   * @return int
 260   */
 261  function discovery_xrds_priority_sort($a, $b) {
 262      if (!isset($a['#priority']) && !isset($b['#priority'])) return 0;
 263      
 264      // if #priority is missing, #priority is assumed to be infinity
 265      if (!isset($a['#priority'])) return 1;
 266      if (!isset($b['#priority'])) return -1;
 267      
 268      if ($a['#priority'] == $b['#priority']) return 0;
 269      return ($a['#priority'] < $b['#priority']) ? -1 : 1;
 270  }

 291   * Obtains the OpenID services for particular identifier by scanning for link
 292   * elements in the returned document.
 293   *
 294   * Note that this function does not use the YADIS protocol to scan for services.
 295   * To use the YADIS protocol, use {@link discovery_get_services()}.
 296   *
 297   * @param string $url the URL
 298   * @return array an array of discovered services, or an empty array if no services
 299   * are found

 301  function discovery_html_get_services($url) {
 302      $services = array();
 303          
 304      $response = http_make_request($url);
 305      $html = $response['data'];
 306          
 307      $uri = _discovery_link_rel('openid2.provider', $html);
 308      $delegate = _discovery_link_rel('openid2.local_id', $html);
 309      
 310      if ($uri) {
 311          $service = array(
 312              'type' => 'http://specs.openid.net/auth/2.0/signon',
 313              'uri' => $uri
 314              );
 315          if ($delegate) $service['localid'] = $delegate;
 316          $services[] = $service;
 317      }
 318  
 319      $uri = _discovery_link_rel('openid.server', $html);
 320      $delegate = _discovery_link_rel('openid.delegate', $html);
 321          
 322      if ($uri) {
 323          $service = array(
 324              'type' => 'http://openid.net/signon/1.0',
 325              'uri' => $uri
 326              );
 327          if ($delegate) $service['localid'] = $delegate;
 328          $services[] = $service;
 329      }
 330      
 331      return $services;
 332  }

 334  /**
 335   * Searches through an HTML document to obtain the value of a meta
 336   * element with a specified http-equiv attribute.
 337   *
 338   * @param string $equiv the http-equiv attribute for which to search
 339   * @param string $html the HTML document to search
 340   * @return mixed the value of the meta element, or FALSE if the element is not
 341   * found
 342   */
 343  function _discovery_meta_httpequiv($equiv, $html) {
 344      $html = preg_replace('/<!(?:--(?:[^-]*|-[^-]+)*--\s*)>/', '', $html); // Strip html comments
 345      
 346      $equiv = preg_quote($equiv);
 347      preg_match('|<meta\s+http-equiv=["\']'. $equiv .'["\'](.*)/?>|iUs', $html, $matches);
 348      if (isset($matches[1])) {
 349          preg_match('|content=["\']([^"]+)["\']|iUs', $matches[1], $content);
 350          if (isset($content[1])) {
 351              return $content[1];
 352          }
 353      }
 354      return FALSE;
 355  }

 357  /**
 358   * Searches through an HTML document to obtain the value of a link
 359   * element with a specified rel attribute.
 360   *
 361   * @param string $rel the rel attribute for which to search
 362   * @param string $html the HTML document to search
 363   * @return mixed the href of the link element, or FALSE if the element is not
 364   * found
 365   */
 366  function _discovery_link_rel($rel, $html) {
 367      $html = preg_replace('/<!(?:--(?:[^-]*|-[^-]+)*--\s*)>/s', '', $html); // Strip html comments
 368      
 369      $rel = preg_quote($rel);
 370      preg_match('|<link\s+rel=["\'](.*)'. $rel .'(.*)["\'](.*)/?>|iUs', $html, $matches);
 371      if (isset($matches[3])) {
 372          preg_match('|href=["\']([^"]+)["\']|iU', $matches[3], $href);
 373          return trim($href[1]);
 374      }
 375      return FALSE;
 376  }

 379   * A simple XRDS parser.
 380   *
 381   * This parser uses the classic expat functions available in PHP to parse the
 382   * XRDS Simple XML document.
 383   *
 384   * The result is an array of discovered services.

 386   * @link http://xrds-simple.net/

 388  class XRDSParser {
 389      /**
 390       * XML parser

 392       * @access private

 394      var $parser;

 411       * CDATA buffer

 413       * @access private
 414       */
 415      var $_buffer;
 416      /**
 417       * Attributes buffer
 418       * @var array
 419       * @access private
 420       */
 421      var $_attribs = array();

 424       * priority attribute buffer

 426       * @access private

 437      /**
 438       * Creates an instance of the XRDS parser.
 439       *
 440       * This constructor also initialises the underlying XML parser.
 441       */
 442      function XRDSParser() {
 443          $this->parser = xml_parser_create_ns();
 444          xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING,0);
 445          xml_set_object($this->parser, $this);
 446          xml_set_element_handler($this->parser, 'element_start', 'element_end');
 447          xml_set_character_data_handler($this->parser, 'cdata');
 448      }
 449      
 450      /**
 451       * Frees memory associated with the underlying XML parser.
 452       *
 453       * Note that only the memory associated with the underlying XML parser is
 454       * freed.  Memory associated with the class itself is not freed.
 455       *
 456       * @access public
 457       */
 458      function free() {
 459          xml_parser_free($this->parser);
 460      }

 462      /**
 463       * Parses an XRDS document.
 464       *
 465       * Once the parsing is complete, use {@link XRDSParser::services()} to obtain
 466       * the services extracted from the document.
 467       *
 468       * @param string $xml the XML document to parse
 469       * @access public
 470       */
 471      function parse($xml) {
 472          xml_parse($this->parser, $xml);
 473      }

 475      /**
 476       * Gets an array of discovered services.
 477       *
 478       * @return array an array of discovered services, or an empty array
 479       * @access public
 480       * @see XRDSParser::parse()

 486      /**
 487       * XML parser callback
 488       *
 489       * @access private
 490       */
 491      function element_start(&$parser, $qualified, $attribs) {
 492          list($ns, $name) = $this->parse_namespace($qualified);
 493  
 494          // Strictly speaking, XML namespace URIs are semi-case sensitive
 495          // (i.e. the scheme and host are not case sensitive, but other elements
 496          // are).  However, the XRDS-Simple specifications defines a
 497          // namespace URI for XRD (xri://$XRD*($v*2.0) rather than xri://$xrd*($v*2.0))
 498          // with an unusual case.
 499          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($name == 'Service')) {
 500              $this->in_service = TRUE;
 501              $this->service = array();
 502              
 503              if (in_array('priority', $attribs)) {
 504                  $this->service['#priority'] = $attribs['priority'];
 505              }
 506              if (in_array('id', $attribs)) {
 507                  $this->service['#id'] = $attribs['id'];
 508              }
 509          }
 510          
 511          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($this->in_service)) {
 512              switch ($name) {
 513                  case 'Type':
 514                  case 'LocalID':
 515                  case 'URI':
 516                      if (in_array('priority', $attribs)) {
 517                          $this->priority = $attribs['priority'];
 518                      } else {
 519                          $this->priority = NULL;
 520                      }
 521              }
 522          }
 523          
 524          $this->_buffer = '';
 525          $this->_attribs = $attribs;
 526      }
 527  
 528      /**
 529       * XML parser callback
 530       *
 531       * @access private
 532       */
 533      function element_end(&$parser, $qualified) {
 534          list($ns, $name) = $this->parse_namespace($qualified);
 535          
 536          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($this->in_service)) {
 537              switch ($name) {
 538                  case 'Service':
 539                      foreach (array('type', 'localid', 'uri') as $key) {
 540                          if (!isset($this->service[$key])) continue;
 541                          $this->service[$key] = $this->flatten_uris($this->service[$key]);
 542                      }
 543                  
 544                      $this->services[] = $this->service;
 545                      $this->in_service = FALSE;
 546                      break;
 547  
 548                  case 'Type':
 549                  case 'LocalID':
 550                  case 'URI':
 551                      $key = strtolower($name);
 552                      if (!isset($this->service[$key])) {
 553                          $this->service[$key] = array();
 554                      }
 555                      if ($this->priority != NULL) {
 556                          $this->service[$key][] = array('#uri' => trim($this->_buffer), '#priority' => $this->priority);
 557                      } else {
 558                          $this->service[$key][] = array('#uri' => trim($this->_buffer));
 559                      }
 560                      $this->priority = NULL;
 561                      break;
 562              }
 563          }
 564          
 565          if ((strtolower($ns) == strtolower(XRD_OPENID_NS)) && ($this->in_service)) {
 566              switch ($name) {
 567                  case 'Delegate':
 568                      $this->service['delegate'] = trim($this->_buffer);
 569              }
 570          }
 571  
 572          $this->_attribs = array();
 573      }

 584      /**
 585       * Parses a namespace-qualified element name.
 586       *
 587       * @param string $qualified the qualified name
 588       * @return array an array with two elements - the first element contains
 589       * the namespace qualifier (or an empty string), the second element contains
 590       * the element name
 591       * @access protected
 592       */
 593      function parse_namespace($qualified) {
 594          $pos = strrpos($qualified, ':');
 595          if ($pos !== FALSE) return array(substr($qualified, 0, $pos), substr($qualified, $pos + 1, strlen($qualified)));
 596          return array('', $qualified);
 597      }

 599      /**
 600       * Flattens the service array.
 601       *
 602       * In an XRDS document, child elements of the service element often contains
 603       * a list of URIs, with the priority specified in the priority attribute.
 604       *
 605       * When the document is parsed in this class, the URI and the priority are first
 606       * extracted into the #uri and the #priority keys respectively.  This function
 607       * takes this array, sorts the elements using the #priority keys (if $sort is
 608       * true), then collapses the array using the value associated with the #uri key.
 609       *
 610       * @param array $array the service array, with URIs and priorities
 611       * @param bool $sort whether to sort the service array using the #priority
 612       * keys
 613       * @return array the services array with URIs sorted by priority
 614       * @access protected
 615       */
 616      function flatten_uris($array, $sort = TRUE) {
 617          $result = array();
 618          
 619          if ($sort) uasort($array, 'discovery_xrds_priority_sort');
 620          
 621          for ($i = 0; $i < count($array); $i++) {
 622              $result[] = $array[$i]['#uri'];
 623          }
 624          
 625          return $result;
 626      }
