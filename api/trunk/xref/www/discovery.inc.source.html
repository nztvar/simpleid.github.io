  72  /**
  73   * Obtains the services for particular identifier.
  74   *
  75   * This function attempts to discover and obtain the XRDS document associated
  76   * with the identifier, parses the XRDS document and returns an array of
  77   * services.
  78   *
  79   * If an XRDS document is not found, and $openid is set to true, this function
  80   * will also attempt to discover OpenID services by looking for link elements
  81   * with rel of openid.server or openid2.provider in the discovered HTML document.
  82   *
  83   * @param string $identifier the identifier
  84   * @param bool $openid if true, performs additional discovery of OpenID services
  85   * by looking for link elements within the discovered document
  86   * @return array an array of discovered services, or an empty array if no services
  87   * are found
  88   */
  89  function discovery_get_services($identifier, $openid = FALSE) {
  90      $identifier = discovery_normalize($identifier);
  91      $url = discovery_get_url($identifier);
  92      
  93      $xrds = discovery_get_xrds($url);
  94  
  95      if ($xrds) {
  96          return discovery_parse_xrds($xrds);
  97      } else {
  98          if ($openid) return discovery_get_openid_services($url);
  99          return array();
 100      }
 101  }

 107   * @param array $services the discovered services
 108   * @param string $type the URI of the type of service to obtain
 109   * @return array an array of matching services, or an empty array of no services

 139  /**
 140   * Obtains a XRDS document at a particular URL.  Performs Yadis discovery if
 141   * the URL does not produce a XRDS document.
 142   *
 143   * @param string $url the URL
 144   * @param bool $check whether to check the content type of the response is
 145   * application/xrds+xml
 146   * @param int $retries the number of tries to make
 147   * @return string the contents of the XRDS document
 148   */
 149  function discovery_get_xrds($url, $check = TRUE, $retries = 5) {
 150      if ($retries == 0) return NULL;
 151      
 152      $response = http_make_request($url, array('Accept' => 'application/xrds+xml'));
 153  
 154      if (isset($response['http-error'])) return NULL;
 155      if (($response['content-type'] == 'application/xrds+xml') || ($check == FALSE)) {
 156          return $response['data'];
 157      } elseif (isset($response['headers']['x-xrds-location'])) {
 158          return discovery_get_xrds($response['headers']['x-xrds-location'], false, $retries - 1);
 159      } else {
 160          $location = _discovery_meta_httpequiv('X-XRDS-Location', $response['data']);
 161          if ($location) {
 162              return discovery_get_xrds($location, false, $retries - 1);
 163          }
 164          return NULL;
 165      }
 166  }

 168  /**
 169   * Normalises an identifier for discovery.
 170   *
 171   * If the identifier begins with xri://, acct: or mailto:, this is stripped out.  If the identifier
 172   * does not begin with a valid URI scheme, http:// is assumed and added to the
 173   * identifier.
 174   *
 175   * @param string $identifier the identifier to normalise
 176   * @return string the normalised identifier
 177   */
 178  function discovery_normalize($identifier) {
 179      $normalized = $identifier;
 180      
 181      if (discovery_is_xri($identifier)) {
 182          if (stristr($identifier, 'xri://') !== false) $normalized = substr($identifier, 6);
 183      } elseif (discovery_is_email($identifier)) {
 184          if (stristr($identifier, 'acct:') !== false) $normalized = substr($identifier, 5);
 185          if (stristr($identifier, 'mailto:') !== false) $normalized = substr($identifier, 7);
 186      } else {
 187          if (stristr($identifier, '://') === false) $normalized = 'http://'. $identifier;
 188          if (substr_count($normalized, '/') < 3) $normalized .= '/';
 189      }
 190      
 191      return $normalized;
 192  }

 213  /**
 214   * Determines whether an identifier is an XRI.
 215   *
 216   * XRI identifiers either start with xri:// or with @, =, +, $ or !.
 217   *
 218   * @param string $identifier the parameter to test
 219   * @return bool true if the identifier is an XRI
 220   */
 221  function discovery_is_xri($identifier) {
 222      $firstchar = substr($identifier, 0, 1);
 223      if ($firstchar == "@" || $firstchar == "=" || $firstchar == "+" || $firstchar == "\$" || $firstchar == "!") return true;
 224      if (stristr($identifier, 'xri://') !== FALSE) return true;
 225      return false;
 226  }

 228  /**
 229   * Determines whether an identifier is an e-mail address.
 230   *
 231   * An identifier is an e-mail address if it:
 232   *
 233   * - has a single @ character
 234   * - does not have a slash character
 235   *
 236   * @param string $identifier the parameter to test
 237   * @return bool true if the identifier is an e-mail address
 238   */
 239  function discovery_is_email($identifier) {
 240      // If it begins with acct: or mailto:, strip it out
 241      if (stristr($identifier, 'acct:') !== false) $identifier = substr($identifier, 5);
 242      if (stristr($identifier, 'mailto:') !== false) $identifier = substr($identifier, 7);
 243      
 244      // If it contains a slash, it is not an e-mail address
 245      if (strpos($identifier, "/") !== false) return false;
 246      
 247      $at = strpos($identifier, "@");
 248      
 249      // If it does not contain a @, it is not an e-mail address
 250      if ($at === false) return false;
 251      
 252      // If it contains more than one @, it is not an e-mail
 253      if (strrpos($identifier, "@") != $at) return false;
 254      
 255      return true;
 256  }

 259  /**
 260   * Callback function to sort service and URI elements based on priorities
 261   * specified in the XRDS document.
 262   *
 263   * The XRDS specification allows multiple instances of certain elements, such
 264   * as Service and URI.  The specification allows an attribute called priority
 265   * so that the document creator can specify the order the elements should be used.
 266   *
 267   * @param array $a
 268   * @param array $b
 269   * @return int
 270   */
 271  function discovery_priority_sort($a, $b) {
 272      if (!isset($a['#priority']) && !isset($b['#priority'])) return 0;
 273      
 274      // if #priority is missing, #priority is assumed to be infinity
 275      if (!isset($a['#priority'])) return 1;
 276      if (!isset($b['#priority'])) return -1;
 277      
 278      if ($a['#priority'] == $b['#priority']) return 0;
 279      return ($a['#priority'] < $b['#priority']) ? -1 : 1;
 280  }

 300  /**
 301   * Obtains the OpenID services for particular identifier by scanning for link
 302   * elements in the returned document.
 303   *
 304   * Note that this function does not use the YADIS protocol to scan for services.
 305   * To use the YADIS protocol, use {@link discovery_get_services()}.
 306   *
 307   * @param string $url the URL
 308   * @return array an array of discovered services, or an empty array if no services
 309   * are found
 310   */
 311  function discovery_get_openid_services($url) {
 312      $services = array();
 313          
 314      $response = http_make_request($url);
 315      $html = $response['data'];
 316          
 317      $uri = _discovery_link_rel('openid2.provider', $html);
 318      $delegate = _discovery_link_rel('openid2.local_id', $html);
 319      
 320      if ($uri) {
 321          $service = array(
 322              'type' => 'http://specs.openid.net/auth/2.0/signon',
 323              'uri' => $uri
 324              );
 325          if ($delegate) $service['localid'] = $delegate;
 326          $services[] = $service;
 327      }
 328  
 329      $uri = _discovery_link_rel('openid.server', $html);
 330      $delegate = _discovery_link_rel('openid.delegate', $html);
 331          
 332      if ($uri) {
 333          $service = array(
 334              'type' => 'http://openid.net/signon/1.0',
 335              'uri' => $uri
 336              );
 337          if ($delegate) $service['localid'] = $delegate;
 338          $services[] = $service;
 339      }
 340      
 341      return $services;
 342  }

 344  /**
 345   * Generates a URI from a URI template and another URI.
 346   *
 347   * The URI must be able to be parsed by PHP's parse_url function.
 348   *
 349   * @param string $template the URI template
 350   * @param string $uri the input URI
 351   * @return string the generated URI
 352   *
 353   */
 354  function discovery_uri_template($template, $uri) {
 355      $parts = @parse_url($uri);
 356      
 357      // The {uri} template variable excludes the fragment
 358      $short_uri = array_shift(explode('#', $uri, 2));
 359      
 360      // The {userinfo} template variable is the user and the pass
 361      $userinfo = (isset($parts['user'])) ? $parts['user'] . ((isset($parts['pass'])) ? ':' . $parts['pass'] : '') : '';
 362      
 363      // The {authority] tempalte is the userinfo, host and port
 364      $authority = $userinfo . ((isset($parts['host'])) ? '@' . $parts['host'] . ((isset($parts['port'])) ? ':' . $parts['port'] : '') : '');
 365      
 366      $args = array(
 367          'uri' => $short_uri,
 368          'scheme' => (isset($parts['scheme'])) ? $parts['scheme'] : '',
 369          'authority' => $authority,
 370          'path' => (isset($parts['path'])) ? $parts['path'] : '',
 371          'query' => (isset($parts['query'])) ? $parts['query'] : '',
 372          'fragment' => (isset($parts['fragment'])) ? $parts['fragment'] : '',
 373          'userinfo' => $userinfo,
 374          'host' => (isset($parts['host'])) ? $parts['host'] : '',
 375          'port' => (isset($parts['port'])) ? $parts['port'] : ''
 376      );
 377      
 378      // Define the % encoded versions
 379      foreach ($args as $key => $value) {
 380          $args['%' . $key] = rfc3986_urlencode($value);
 381      }
 382      
 383      return strtr($template, $args);
 384  }

 386  /**
 387   * Searches through an HTML document to obtain the value of a meta
 388   * element with a specified http-equiv attribute.
 389   *
 390   * @param string $equiv the http-equiv attribute for which to search
 391   * @param string $html the HTML document to search
 392   * @return mixed the value of the meta element, or FALSE if the element is not
 393   * found
 394   */
 395  function _discovery_meta_httpequiv($equiv, $html) {
 396      $html = preg_replace('/<!(?:--(?:[^-]*|-[^-]+)*--\s*)>/', '', $html); // Strip html comments
 397      
 398      $equiv = preg_quote($equiv);
 399      preg_match('|<meta\s+http-equiv=["\']'. $equiv .'["\'](.*)/?>|iUs', $html, $matches);
 400      if (isset($matches[1])) {
 401          preg_match('|content=["\']([^"]+)["\']|iUs', $matches[1], $content);
 402          if (isset($content[1])) {
 403              return $content[1];
 404          }
 405      }
 406      return FALSE;
 407  }

 409  /**
 410   * Searches through an HTML document to obtain the value of a link
 411   * element with a specified rel attribute.
 412   *
 413   * @param string $rel the rel attribute for which to search
 414   * @param string $html the HTML document to search
 415   * @return mixed the href of the link element, or FALSE if the element is not
 416   * found
 417   */
 418  function _discovery_link_rel($rel, $html) {
 419      $html = preg_replace('/<!(?:--(?:[^-]*|-[^-]+)*--\s*)>/s', '', $html); // Strip html comments
 420      
 421      $rel = preg_quote($rel);
 422      preg_match('|<link\s+rel=["\'](.*)'. $rel .'(.*)["\'](.*)/?>|iUs', $html, $matches);
 423      if (isset($matches[3])) {
 424          preg_match('|href=["\']([^"]+)["\']|iU', $matches[3], $href);
 425          return trim($href[1]);
 426      }
 427      return FALSE;
 428  }

 609      /**
 610       * XML parser callback
 611       *
 612       * @access private
 613       */
 614      function element_start(&$parser, $qualified, $attribs) {
 615          list($ns, $name) = $this->parse_namespace($qualified);
 616  
 617          // Strictly speaking, XML namespace URIs are semi-case sensitive
 618          // (i.e. the scheme and host are not case sensitive, but other elements
 619          // are).  However, the XRDS-Simple specifications defines a
 620          // namespace URI for XRD (xri://$XRD*($v*2.0) rather than xri://$xrd*($v*2.0))
 621          // with an unusual case.
 622          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($name == 'Service')) {
 623              $this->in_service = TRUE;
 624              $this->service = array();
 625              
 626              if (in_array('priority', $attribs)) {
 627                  $this->service['#priority'] = $attribs['priority'];
 628              }
 629              if (in_array('id', $attribs)) {
 630                  $this->service['#id'] = $attribs['id'];
 631              }
 632          }
 633          if (($ns == XRD_NS) && ($name == 'Link')) {
 634              $this->in_service = TRUE;
 635              $this->service = array();
 636              
 637              foreach ($attribs as $name => $value) {
 638                  $this->service[(in_array($name, $this->xrds_map)) ? $this->xrds_map[$name] : $name] = $value;
 639              }
 640          }
 641          
 642          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($this->in_service)) {
 643              switch ($name) {
 644                  case 'Type':
 645                  case 'LocalID':
 646                  case 'URI':
 647                      if (in_array('priority', $attribs)) {
 648                          $this->priority = $attribs['priority'];
 649                      } else {
 650                          $this->priority = NULL;
 651                      }
 652              }
 653          }
 654          if (strtolower($ns) == strtolower(XRD_NS)) {
 655              $this->priority = NULL;
 656          }
 657          
 658          $this->_buffer = '';
 659          $this->_attribs = $attribs;
 660      }

 662      /**
 663       * XML parser callback
 664       *
 665       * @access private
 666       */
 667      function element_end(&$parser, $qualified) {
 668          list($ns, $name) = $this->parse_namespace($qualified);
 669          
 670          if (($ns == XRD_NS) && !$this->in_service) {
 671              switch ($name) {
 672                  case 'Subject':
 673                      $this->subject = trim($this->_buffer);
 674                      break;
 675                  case 'Alias':
 676                      $this->aliases[] = trim($this->_buffer);
 677                      break;
 678                  case 'Property':
 679                      $this->properties[$this->_attribs['type']] = trim($this->_buffer);
 680                      break;
 681              }
 682          }
 683          
 684          if ((strtolower($ns) == strtolower(XRD2_NS)) && ($this->in_service)) {
 685              switch ($name) {
 686                  case 'Service':
 687                      foreach (array('type', 'localid', 'uri') as $key) {
 688                          if (!isset($this->service[$key])) continue;
 689                          $this->service[$key] = $this->flatten_uris($this->service[$key]);
 690                      }
 691                  
 692                      $this->services[] = $this->service;
 693                      $this->in_service = FALSE;
 694                      break;
 695  
 696                  case 'Type':
 697                  case 'LocalID':
 698                  case 'URI':
 699                      $key = strtolower($name);
 700                      if (!isset($this->service[$key])) {
 701                          $this->service[$key] = array();
 702                      }

 704                          $this->service[$key][] = array('#uri' => trim($this->_buffer), '#priority' => $this->priority);
 705                      } else {

 709                      break;

 711          }

 714                  case 'Link':

 716                      $this->in_service = FALSE;
 717                      break;
 718                  case 'Property':
 719                      if (!isset($this->service['properties'])) $this->service['properties'] = array();
 720                      $this->service['properties'][$this->_attribs['type']] = trim($this->_buffer);
 721                  case 'Title':
 722                      $this->service['title'] = trim($this->_buffer);
 723                      break;
 724              }
 725          }
 726

 728              switch ($name) {
 729                  case 'Delegate':
 730                      $this->service['delegate'] = trim($this->_buffer);
 731              }
 732          }
 733  
 734          $this->_attribs = array();

 761      /**
 762       * Flattens the service array.
 763       *
 764       * In an XRDS document, child elements of the service element often contains
 765       * a list of URIs, with the priority specified in the priority attribute.
 766       *
 767       * When the document is parsed in this class, the URI and the priority are first
 768       * extracted into the #uri and the #priority keys respectively.  This function
 769       * takes this array, sorts the elements using the #priority keys (if $sort is
 770       * true), then collapses the array using the value associated with the #uri key.
 771       *
 772       * @param array $array the service array, with URIs and priorities
 773       * @param bool $sort whether to sort the service array using the #priority
 774       * keys
 775       * @return array the services array with URIs sorted by priority
 776       * @access protected
 777       */
 778      function flatten_uris($array, $sort = TRUE) {
 779          $result = array();
 780          
 781          if ($sort) uasort($array, 'discovery_priority_sort');
 782          
 783          for ($i = 0; $i < count($array); $i++) {
 784              $result[] = $array[$i]['#uri'];
 785          }
 786          
 787          return $result;
