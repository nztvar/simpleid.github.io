 939      /**
 940       * generates the array containing to-be-parsed stuff:
 941       * $blocks["main"],$blocks["main.table"],$blocks["main.table.row"], etc.
 942       * also builds the reverse parse order.
 943       *
 944       * @access public - aiming for private
 945       * @param string $con content to be processed
 946       * @param string $parentblock name of the parent block in the block hierarchy
 947       */
 948  	function _maketree ($con, $parentblock='') {
 949  
 950          $blocks = array();
 951  
 952          $con2 = explode($this->block_start_delim, $con);
 953  
 954          if (!empty($parentblock)) {
 955  
 956              $block_names = explode('.', $parentblock);
 957              $level = sizeof($block_names);
 958  
 959          } else {
 960  
 961              $block_names = array();
 962              $level = 0;
 963          }
 964  
 965          // JRC 06/04/2005 Added block comments (on BEGIN or END) <!-- BEGIN: block_name#Comments placed here -->
 966          //$patt = "($this->block_start_word|$this->block_end_word)\s*(\w+)\s*$this->block_end_delim(.*)";
 967          $patt = "(" . $this->block_start_word . "|" . $this->block_end_word . ")\s*(\w+)" . $this->comment_preg . "\s*" . $this->block_end_delim . "(.*)";
 968  
 969          foreach($con2 as $k => $v) {
 970  
 971              $res = array();
 972  
 973              if (preg_match_all("/$patt/ims", $v, $res, PREG_SET_ORDER)) {
 974                  // $res[0][1] = BEGIN or END
 975                  // $res[0][2] = block name
 976                  // $res[0][3] = comment
 977                  // $res[0][4] = kinda content
 978                  $block_word    = $res[0][1];
 979                  $block_name    = $res[0][2];
 980                  $comment    = $res[0][3];
 981                  $content    = $res[0][4];
 982  
 983                  if (strtoupper($block_word) == $this->block_start_word) {
 984  
 985                      $parent_name = implode('.', $block_names);
 986  
 987                      // add one level - array("main","table","row")
 988                      $block_names[++$level] = $block_name;
 989  
 990                      // make block name (main.table.row)
 991                      $cur_block_name=implode('.', $block_names);
 992  
 993                      // build block parsing order (reverse)
 994                      $this->block_parse_order[] = $cur_block_name;
 995  
 996                      //add contents. trinary operator eliminates assign error in E_ALL reporting
 997                      $blocks[$cur_block_name] = isset($blocks[$cur_block_name]) ? $blocks[$cur_block_name] . $content : $content;
 998  
 999                      // add {_BLOCK_.blockname} string to parent block
1000                      $blocks[$parent_name] .= str_replace('\\', '', $this->tag_start_delim) . '_BLOCK_.' . $cur_block_name . str_replace('\\', '', $this->tag_end_delim);
1001  
1002                      // store sub block names for autoresetting and recursive parsing
1003                      $this->sub_blocks[$parent_name][] = $cur_block_name;
1004  
1005                      // store sub block names for autoresetting
1006                      $this->sub_blocks[$cur_block_name][] = '';
1007  
1008                  } else if (strtoupper($block_word) == $this->block_end_word) {
1009  
1010                      unset($block_names[$level--]);
1011  
1012                      $parent_name = implode('.', $block_names);
1013  
1014                      // add rest of block to parent block
1015                      $blocks[$parent_name] .= $content;
1016                  }
1017              } else {
1018  
1019                  // no block delimiters found
1020                  // Saves doing multiple implodes - less overhead
1021                  $tmp = implode('.', $block_names);
1022  
1023                  if ($k) {
1024                      $blocks[$tmp] .= $this->block_start_delim;
1025                  }
1026  
1027                  // trinary operator eliminates assign error in E_ALL reporting
1028                  $blocks[$tmp] = isset($blocks[$tmp]) ? $blocks[$tmp] . $v : $v;
1029              }
1030          }
1031  
1032          return $blocks;
1033      }

1153  
1154              /**
1155               * Support hierarchy of file locations to search
1156               *
1157               * @example Supply array of filepaths when instantiating
1158               *             First path supplied that has the named file is prioritised
1159               *             $xtpl = new XTemplate('myfile.xtpl', array('.','/mypath', '/mypath2'));

1172

1174              }

1176

1178

1180

1182

1190
