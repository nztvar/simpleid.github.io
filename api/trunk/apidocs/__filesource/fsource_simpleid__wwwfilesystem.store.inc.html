  24. /**
  25.  * Functions for persistent storage via the file system.
  26.  *
  27.  * In general, there are three different sets of data which SimpleID needs
  28.  * to store:
  29.  *
  30.  * - transient data (e.g. OpenID associations, sessions, auto-login)
  31.  * - application data (e.g. salt for form tokens)
  32.  * - user data (e.g. user names, passwords and settings)
  33.  *
  34.  * Prior to version 0.7, both transient data and application data are stored
  35.  * using {@link cache.inc}.  From version 0.7, application data are now
  36.  * stored separately from the cache.
  37.  *
  38.  * Prior to version 0.7, user data is only stored in the identity file, to which
  39.  * SimpleID cannot write.  This means that if the user wishes to change a setting,
  40.  * he or she will need to edit the identity file manually.  Other user settings
  41.  * (e.g. RP preferences) are stored using {@link cache.inc}
  42.  *
  43.  * From version 0.7, user data is stored in two files, one is the identity
  44.  * file, the other is the user store file, which SimpleID can write.
  45.  *
  46.  * @package simpleid
  47.  * @filesource
  48.  */
  49.

  64. function store_user_exists($uid) {

  69.         return false;

  73. /**
  74.  * Loads user data for a specified user name.
  75.  *
  76.  * The user name must exist.  You should check whether the user name exists with
  77.  * the {@link store_user_exists()} function
  78.  *
  79.  * @param string $uid the name of the user to load
  80.  * @return mixed data for the specified user
  81.  */
  82. function store_user_load($uid) {
  83.     if (!_store_is_valid_name($uid)) return array();
  84.     $store_file = SIMPLEID_STORE_DIR . "/$uid.usrstore";
  85.     
  86.     if (file_exists($store_file)) {
  87.         $data = unserialize(file_get_contents($store_file));
  88.     } else {
  89.         $data = array();
  90.     }
  91.     
  92.     $identity_file = SIMPLEID_IDENTITIES_DIR . "/$uid.identity";
  93.     $data = array_merge($data, parse_ini_file($identity_file, TRUE));
  94.     
  95.     return $data;
  96. }

  98. /**
  99.  * Verifies a set of credentials for a specified user.
  100.  *
  101.  * A set of credentials comprises:
  102.  *
  103.  * - A user name
  104.  * - Some kind of verifying information, such as a plaintext password, a hashed
  105.  *   password (e.g. digest) or some other kind of identifying information.
  106.  *
  107.  * The user name is passed to this function using the $uid parameter.  The user
  108.  * name may or may not exist.  If the user name does not exist, this function
  109.  * <strong>must</strong> return false.
  110.  *
  111.  * The credentials are supplied as an array using the $credentials parameter.
  112.  * Typically this array will be a subset of the $_POST superglobal passed to the
  113.  * {@link user_login()} function.  Thus it will generally contain the keys 'pass' and
  114.  * 'digest'.
  115.  *
  116.  * This function must check whether the credentials supplied matches the credentials
  117.  * for the specified user in the store.  If for any reason that credentials
  118.  * do not match, this function <strong>must</strong> return false.
  119.  *
  120.  * @param string $uid the name of the user to verify
  121.  * @param array $credentials the credentials supplied by the browser
  122.  * @param bool $allow_legacy_login true to verify using a plaintext password instead of
  123.  *  digest
  124.  * @return bool whether the credentials supplied matches those for the specified
  125.  *  user
  126.  */
  127. function store_user_verify_credentials($uid, $credentials, $allow_legacy_login = FALSE) {
  128.     $test_user = user_load($uid);
  129.     
  130.     if ($test_user == NULL) return false;
  131.     
  132.     if ($credentials['digest'] && !_user_verify_digest($credentials['digest'], $credentials['nonce'], $test_user)) {
  133.         return false;
  134.     } elseif ($allow_legacy_login && !$credentials['digest'] && (md5($credentials['pass']) != $test_user['pass'])) {
  135.         return false;
  136.     }
  137.     
  138.     return true;
  139. }

  141. /**
  142.  * Finds the user name from a specified OpenID Identity URI.
  143.  *
  144.  * @param string $identity the Identity URI of the user to load
  145.  * @return string the user name for the Identity URI, or NULL if no user has
  146.  *  the specified Identity URI
  147.  */
  148. function store_get_uid($identity) {
  149.     $uid = cache_get('identity', $identity);
  150.     if ($uid !== NULL) return $uid;
  151.     
  152.     $r = NULL;
  153.     
  154.     $dir = opendir(SIMPLEID_IDENTITIES_DIR);
  155.     
  156.     while (($file = readdir($dir)) !== false) {
  157.         $filename = SIMPLEID_IDENTITIES_DIR . '/' . $file;
  158.         
  159.         if ((filetype($filename) != "file") || (!preg_match('/^(.+)\.identity$/', $file, $matches))) continue;
  160.         
  161.         $uid = $matches[1];
  162.         $test_user = store_user_load($uid);
  163.         
  164.         cache_set('identity', $test_user['identity'], $uid);
  165.     
  166.         if ($test_user['identity'] == $identity) {
  167.             $r = $uid;
  168.         }
  169.     }
  170.         
  171.     closedir($dir);
  172.     
  173.     return $r;
  174. }

  176. /**
  177.  * Saves user data for a specific user name.
  178.  *
  179.  * This data is stored in the user store file.
  180.  *
  181.  * @param string $uid the name of the user
  182.  * @param array $data the data to save
  183.  * @param array $exclude an array of keys to exclude from the user store file.
  184.  *  These are generally keys which are stored in the identity file.
  185.  *
  186.  * @since 0.7
  187.  */
  188. function store_user_save($uid, $data, $exclude = array()) {
  189.     foreach ($exclude as $key) {
  190.         if (isset($data[$key])) unset($data[$key]);
  191.     }
  192.     
  193.     if (!_store_is_valid_name($uid)) {
  194.         trigger_error("Invalid user name for filesystem store", E_USER_ERROR);
  195.         return;
  196.     }
  197.     
  198.     $store_file = SIMPLEID_STORE_DIR . "/$uid.usrstore";
  199.     $file = fopen($store_file, 'w');
  200.     fwrite($file, serialize($data));
  201.     fclose($file);
  202. }

  204. /**
  205.  * Loads an application setting.
  206.  *
  207.  * @param string $name the name of the setting to return
  208.  * @param mixed $default the default value to use if this variable has never been set
  209.  * @return mixed the value of the setting
  210.  *
  211.  */
  212. function store_get($name, $default = NULL) {
  213.     global $simpleid_settings;
  214.     
  215.     if (!_store_is_valid_name($name)) return $default;
  216.     
  217.     if (!isset($simpleid_settings[$name])) {
  218.         $setting_file = SIMPLEID_STORE_DIR . "/$name.setting";
  219.         
  220.         if (file_exists($setting_file)) {
  221.             $simpleid_settings[$name] = unserialize(file_get_contents($setting_file));
  222.         } else {
  223.             return $default;
  224.         }
  225.     }
  226.     
  227.     return $simpleid_settings[$name];
  228. }

  231.  * Saves an application setting.
  232.  *
  233.  * @param string $name the name of the setting to save
  234.  * @param mixed $value the value of the setting
  235.  *
  236.  */
  237. function store_set($name, $value) {

  239.     
  240.     if (!_store_is_valid_name($name)) {
  241.         trigger_error("Invalid setting name for filesystem store", E_USER_ERROR);

  246.     
  247.     $setting_file = SIMPLEID_STORE_DIR . "/$name.setting";

  249.     fwrite($file, serialize($value));
  250.     fclose($file);
  251. }

  253. /**
  254.  * Deletes an application setting.
  255.  *
  256.  * @param string $name the name of the setting to delete
  257.  *
  258.  */
  259. function store_del($name) {
  260.     global $simpleid_settings;
  261.     
  262.     if (!_store_is_valid_name($name)) {
  263.         trigger_error("Invalid setting name for filesystem store", E_USER_ERROR);
  264.         return;
  265.     }
  266.     
  267.     if (isset($simpleid_settings[$name])) unset($simpleid_settings[$name]);
  268.     
  269.     $setting_file = SIMPLEID_STORE_DIR . "/$name.setting";

  273. /**
  274.  * Determines whether a name is a valid name for use with this store.
  275.  *
  276.  * For file system storage, a name is not valid if it contains either a
  277.  * directory separator (i.e. / or \).
  278.  *
  279.  * @param string $name the name to check
  280.  * @return boolean whether the name is valid for use with this store
  281.  *
  282.  */
  283. function _store_is_valid_name($name) {
  284.     return preg_match('!\A[^/\\\\]*\z!', $name);
  285. }
