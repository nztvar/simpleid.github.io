  37. /**
  38.  * Performs an HTTP request.
  39.  *
  40.  * Communication with the web server is conducted using libcurl where possible.
  41.  * Where libcurl does not exist, then sockets will be used.
  42.  *
  43.  * @param string $url the URL
  44.  * @param array $headers HTTP headers containing name => value pairs
  45.  * @param string $method the HTTP request method
  46.  * @param int $retry the maximum number of redirects allowed
  47.  * @return array containing keys 'error-code' (for communication errors), 'error'
  48.  *  (for communication errors), 'data' (content returned), 'code' (the HTTP status code), 'http-error'
  49.  *  (if the HTTP status code is not 200 or 304), 'protocol' (the HTTP protocol in the response),
  50.  *  'headers' (an array of return headers in lowercase),
  51.  *  'content-type' (the HTTP content-type returned)
  52.  */
  53. function http_make_request($url, $headers = array(), $method = 'GET', $retry = 3) {
  54.     // If CURL is available, we use it
  55.     if (extension_loaded('curl')) {
  56.         $response = _http_make_request_curl($url, $headers, $method, $retry);
  57.     } else {
  58.         $response = _http_make_request_fsock($url, $headers, $method, $retry);
  59.     }
  60.     
  61.     if (!isset($response['error-code'])) {
  62.         $valid_codes = array(
  63.             100, 101,
  64.             200, 201, 202, 203, 204, 205, 206,
  65.             300, 301, 302, 303, 304, 305, 307,
  66.             400, 401, 402, 403, 404, 405, 406, 407, 408, 409, 410, 411, 412, 413, 414, 415, 416, 417,
  67.             500, 501, 502, 503, 504, 505
  68.         );
  69.     
  70.         // RFC 2616 states that all unknown HTTP codes must be treated the same as the
  71.         // base code in their class.
  72.         if (!in_array($response['code'], $valid_codes)) {
  73.             $response['code'] = floor($response['code'] / 100) * 100;
  74.         }
  75.         
  76.         if (($response['code'] != 200) && ($response['code'] != 304)) {
  77.             $response['http-error'] = $response['code'];
  78.         }
  79.         
  80.     }
  81.  
  82.     return $response;
  83. }

  103. /**
  104.  * Performs an HTTP request using libcurl.
  105.  *
  106.  * @param string $url the URL
  107.  * @param array $headers HTTP headers containing name => value pairs
  108.  * @param string $method the HTTP request method
  109.  * @param int $retry the maximum number of redirects allowed
  110.  * @return array containing keys 'error-code' (for communication errors), 'error'
  111.  *  (for communication errors), 'data' (content returned), 'code' (the HTTP status code), 'http-error'
  112.  *  (if the HTTP status code is not 200 or 304), 'headers' (an array of return headers),
  113.  *  'content-type' (the HTTP content-type returned)
  114.  */
  115. function _http_make_request_curl($url, $headers = array(), $method = 'GET', $retry = 3) {
  116.     // CURLOPT_FOLLOWLOCATION only works when safe mode is off or when open_basedir is set
  117.     // In these instances we will need to follow redirects manually
  118.     $manual_redirect = ((@ini_get('safe_mode') === 1)   // safe mode
  119.         || (strtolower(@ini_get('safe_mode')) == 'on')  // safe mode
  120.         || (@ini_get('open_basedir') != false)); // open_basedir
  121.     
  122.     $version = curl_version();
  123.     
  124.     $curl = curl_init($url);
  125.     
  126.     if (version_compare($version['version'], '7.10.5', '>=')) {
  127.         curl_setopt($curl, CURLOPT_ENCODING, '');
  128.     }
  129.     
  130.     if (!$manual_redirect) curl_setopt($curl, CURLOPT_FOLLOWLOCATION, true);
  131.     
  132.     curl_setopt($curl, CURLOPT_CUSTOMREQUEST, $method);
  133.     curl_setopt($curl, CURLOPT_MAXREDIRS, $retry);
  134.     curl_setopt($curl, CURLOPT_HTTPHEADER, array(implode("\n", $headers) . "\n"));
  135.     curl_setopt($curl, CURLOPT_USERAGENT, SIMPLEHTTP_USER_AGENT);
  136.     
  137.     curl_setopt($curl, CURLOPT_TIMEOUT, 20);
  138.     curl_setopt($curl, CURLOPT_CONNECTTIMEOUT, 20);
  139.     
  140.     curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
  141.     curl_setopt($curl, CURLOPT_HEADER, true);
  142.     
  143.     curl_setopt($curl, CURLOPT_SSL_VERIFYPEER, false);
  144.     curl_setopt($curl, CURLOPT_SSL_VERIFYHOST, false);
  145.     
  146.     $response = curl_exec($curl);
  147.     
  148.     if (($response === FALSE) && ((curl_errno($curl) == 23) || (curl_errno($curl) == 61))) {
  149.         curl_setopt($curl, CURLOPT_ENCODING, 'none');
  150.         $response = curl_exec($curl);
  151.     }
  152.     
  153.     if ($response === FALSE) {
  154.         $result = array();
  155.         $result['error-code'] = curl_errno($curl);
  156.         $result['error'] = curl_error($curl);
  157.     } else {
  158.         $result['code'] = curl_getinfo($curl, CURLINFO_HTTP_CODE);
  159.         $result['url'] = curl_getinfo($curl, CURLINFO_EFFECTIVE_URL);
  160.         $result['content-type'] = curl_getinfo($curl, CURLINFO_CONTENT_TYPE);
  161.         
  162.         // Parse response.
  163.         $result['raw'] = $response;
  164.         
  165.         $header_size = curl_getinfo($curl, CURLINFO_HEADER_SIZE);
  166.         $result['data'] = substr($response, $header_size);
  167.         
  168.         $response_headers = substr($response, 0, $header_size - 4);
  169.  
  170.         // In case where redirect occurs, we want the last set of headers
  171.         $header_blocks = explode("\r\n\r\n", $response_headers);
  172.         $header_block = array_pop($header_blocks);
  173.  
  174.         $result = array_merge($result, _http_parse_headers($header_block, TRUE));
  175.         
  176.         // If we are in safe mode, we need to process redirects manually
  177.         if ($manual_redirect && (($result['code'] == 301) || ($result['code'] == 302) || ($result['code'] == 307))) {
  178.             if ($retry == 0) {
  179.                 // Too many times, return an error
  180.                 $result['error-code'] = 47;
  181.                 $result['error'] = 'Too many redirects';
  182.             } else {
  183.                 curl_close($curl);
  184.                 return _http_make_request_curl($result['headers']['location'], $headers, $method, $retry - 1);
  185.             }
  186.         }
  187.     }
  188.     
  189.     curl_close($curl);
  190.  
  191.     return $result;
  192. }

  194. /**
  195.  * Performs an HTTP request using sockets.
  196.  *
  197.  * @param string $url the URL
  198.  * @param array $headers HTTP headers containing name => value pairs
  199.  * @param string $method the HTTP request method
  200.  * @param int $retry the maximum number of redirects allowed
  201.  * @return array containing keys 'error-code' (for communication errors), 'error'
  202.  *  (for communication errors), 'data' (content returned), 'code' (the HTTP status code), 'http-error'
  203.  *  (if the HTTP status code is not 200 or 304), 'headers' (an array of return headers),
  204.  *  'content-type' (the HTTP content-type returned)
  205.  */
  206. function _http_make_request_fsock($url, $headers = array(), $method = 'GET', $retry = 3) {
  207.     $result = array();
  208.     
  209.     $parts = parse_url($url);
  210.     
  211.     if (!isset($parts)) {
  212.         $result['error-code'] = 3;
  213.         $result['error'] = 'URL not properly formatted';
  214.         return $result;
  215.     }
  216.     
  217.     if ($parts['scheme'] == 'http') {
  218.         $port = isset($parts['port']) ? $parts['port'] : 80;
  219.         $host = $parts['host'];
  220.     } elseif ($parts['scheme'] == 'https') {
  221.         $port = isset($parts['port']) ? $parts['port'] : 443;
  222.         $host = 'ssl://' . $uri['host'];
  223.     } else {
  224.         $result['error-code'] = 1;
  225.         $result['error'] = 'Unsupported protocol';
  226.     }
  227.     
  228.     $fp = @fsockopen($host, $port, $errno, $errstr, 15);
  229.  
  230.     if (!$fp) {
  231.         $result['error-code'] = 7;
  232.         $result['error'] = "Cannot connect: Error $errno:" . trim($errstr);
  233.         return $result;
  234.     }
  235.     
  236.     if (isset($parts['path'])) {
  237.         $path = $url_parts['path'];
  238.         if (isset($parts['query'])) $path .= '?' . $url_parts['query'];
  239.     } else {
  240.         $path = '/';
  241.     }
  242.  
  243.     $headers = array_merge(
  244.         array(
  245.             'Host' => $parts['host'],
  246.             'User-Agent' => SIMPLEHTTP_USER_AGENT,
  247.             'Connection' => 'close'
  248.         ),
  249.         $headers
  250.     );
  251.  
  252.     if (isset($parts['user']) && isset($parts['pass'])) {
  253.         $headers['Authorization'] = 'Basic '. base64_encode($uri['user'] . (!empty($uri['pass']) ? ":". $uri['pass'] : ''));
  254.     }
  255.   
  256.     $request = $method . ' '. $path ." HTTP/1.0\r\n";
  257.     
  258.     $keys = array_keys($headers);
  259.     for ($i = 0; $i < count($keys); $i++) {
  260.         $request .= $keys[$i] . ': ' . $headers[$keys[$i]] . "\r\n";
  261.     }
  262.     
  263.     // End of headers - separator
  264.     $request .= "\r\n";
  265.     
  266.     fwrite($fp, $request);
  267.  
  268.     // Fetch response.
  269.     $response = '';
  270.     while (!feof($fp) && $chunk = fread($fp, 1024)) {
  271.         $response .= $chunk;
  272.     }
  273.     fclose($fp);
  274.  
  275.     // Parse response.
  276.     list($header_block, $result['data']) = explode("\r\n\r\n", $response, 2);
  277.     
  278.     $result = array_merge($result, _http_parse_headers($header_block, FALSE));
  279.     
  280.     // Process redirects
  281.     if (($result['code'] == 301) || ($result['code'] == 302) || ($result['code'] == 307)) {
  282.         if ($retry == 0) {
  283.             // Too many times, return an error
  284.             $result['error-code'] = 47;
  285.             $result['error'] = 'Too many redirects';
  286.         } else {
  287.             $result = _http_make_request_fsock($result['headers']['location'], $headers, $method, $retry - 1);
  288.         }
  289.     }
  290.  
  291.     $result['url'] = $url;
  292.     return $result;
  293. }

  295. /**
  296.  * Parses HTTP response headers.
  297.  *
  298.  * @param string $header_block the unparsed header block
  299.  * @param bool $curl if true, use simplified parsing as libcurl already parses
  300.  *  the headers
  301.  * @return an array containing the following keys: 'protocol' (the HTTP protocol in the response),
  302.  *  'headers' (an array of return headers in lowercase).  If $curl is false, additional
  303.  *  parsing is done for 'code' and 'content-type'
  304.  */
  305. function _http_parse_headers($header_block, $curl) {
  306.     $headers = array();
  307.     $result = array();
  308.     
  309.     // Split the status line from the rest of the message header
  310.     list($status, $header_block) = preg_split("/\r\n|\n|\r/", $header_block, 2);
  311.     
  312.     // RFC 2616, section 4.2: Header fields can be extended over multiple lines
  313.     // by preceding each extra line with at least one space or tab.  So we need
  314.     // to join them...
  315.     $header_block = preg_replace('/(\r\n|\n|\r)( |\t)+/', '', $header_block);
  316.     
  317.     // Then split them to get the fields
  318.     $fields = preg_split("/\r\n|\n|\r/", $header_block);
  319.     
  320.     // Parse the status line
  321.     list($protocol, $code, $reason) = explode(' ', trim($status), 3);
  322.     
  323.     $result['protocol'] = $protocol;    
  324.     if (!$curl) $result['code'] = $code;
  325.  
  326.     // Parse headers.
  327.     while ($field = trim(array_shift($fields))) {
  328.         list($header, $value) = explode(':', $field, 2);
  329.         
  330.         // Headers are case insensitive
  331.         $header = strtolower($header);
  332.         
  333.         if (isset($headers[$header])) {
  334.             // RFC 2616, section 4.2: Multiple headers with the same field
  335.             // name is the same as a concatenating all the headers in a single
  336.             // header, separated by commas.
  337.             $headers[$header] .= ','. trim($value);
  338.         } else {
  339.             $headers[$header] = trim($value);
  340.         }
  341.         
  342.         if (!$curl && (strtolower($header) == 'content-type')) $result['content-type'] = $value;
  343.     }
  344.         
  345.     $result['headers'] = $headers;
  346.     return $result;
  347. }
