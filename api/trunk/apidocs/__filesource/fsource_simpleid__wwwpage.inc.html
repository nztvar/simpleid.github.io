  88. /**
  89.  * Returns the user's home page.
  90.  */
  91. function page_sites() {
  92.     global $user;
  93.     global $xtpl;
  94.     
  95.     if ($user == NULL) {
  96.         user_login_form('my/sites');
  97.         return;
  98.     }
  99.     
  100.     user_header();
  101.     page_nav();
  102.     
  103.     if (isset($user['rp'])) {
  104.         $user_rps =& $user['rp'];
  105.     } else {
  106.         $user_rps = array();
  107.     }
  108.     
  109.     if (isset($_POST['tk'])) {
  110.         if (!validate_form_token($_POST['tk'], 'autorelease')) {
  111.             set_message('SimpleID detected a potential security attack.  Please try again.');
  112.         } else {
  113.             if (isset($_POST['autorelease'])) {
  114.                 foreach ($_POST['autorelease'] as $realm => $autorelease) {
  115.                     if (isset($user_rps[$realm])) {
  116.                         $user_rps[$realm]['auto_release'] = ($autorelease) ? 1 : 0;
  117.                     }
  118.                 }
  119.             }
  120.             
  121.             if (isset($_POST['remove'])) {
  122.                 foreach ($_POST['remove'] as $realm => $autorelease) {
  123.                     if (isset($user_rps[$realm])) {
  124.                         unset($user_rps[$realm]);
  125.                     }
  126.                 }
  127.             }
  128.             
  129.             if (isset($_POST['update-all'])) {
  130.                 foreach ($user_rps as $realm => $values) {
  131.                     $user_rps[$realm]['auto_release'] = (isset($_POST['autorelease'][$realm]) && $_POST['autorelease'][$realm]) ? 1 : 0;
  132.                 }
  133.             }
  134.             
  135.             user_save($user);
  136.     
  137.             set_message('Your preferences have been saved.');
  138.         }
  139.     }
  140.     
  141.     if ($user_rps) {
  142.         foreach ($user_rps as $realm => $rp) {
  143.             $xtpl->assign('realm', htmlspecialchars($rp['realm'], ENT_QUOTES, 'UTF-8'));
  144.             $xtpl->assign('last_time', htmlspecialchars($rp['last_time'], ENT_QUOTES, 'UTF-8'));
  145.             $xtpl->assign('last_time_formatted', htmlspecialchars(strftime(SIMPLEID_DATE_TIME_FORMAT, $rp['last_time']), ENT_QUOTES, 'UTF-8'));
  146.             $xtpl->assign('auto_release', (isset($rp['auto_release']) && $rp['auto_release']) ? 'checked="checked"' : '');
  147.             
  148.             if (SIMPLEID_VERIFY_RETURN_URL_USING_REALM) {
  149.                 // $rp_info would usually expire by now, so we allow for stale results to be retrieved to improve performance
  150.                 $rp_info = simpleid_get_rp_info($realm, TRUE);
  151.                 if (!isset($rp_info['return_to_verified']) || !$rp_info['return_to_verified']) $xtpl->assign('realm_class', 'return-to-suspect');
  152.             }
  153.             
  154.             $xtpl->parse('main.sites.realm');
  155.         }
  156.     }
  157.     
  158.     if (!$user_rps || (count($user_rps) == 0)) {
  159.         $xtpl->assign('disabled', 'disabled="disabled"');
  160.     }
  161.     
  162.     $xtpl->assign('token', get_form_token('autorelease'));
  163.         
  164.     $xtpl->parse('main.sites');
  165.     
  166.     $xtpl->assign('title', 'My Sites');
  167.     $xtpl->assign('javascript', '<script src="' . get_base_path() . 'html/openid-consent.js" type="text/javascript"></script>');
  168.     $xtpl->parse('main');
  169.     $xtpl->out('main');
  170. }

  188. /**
  189.  * Renders a particular block.
  190.  *
  191.  * @param array $block the block to render
  192.  * @return string the HTML of the rendered block
  193.  */
  194. function page_render_block($block) {
  195.     static $xtpl_block;
  196.     
  197.     if (!$xtpl_block) $xtpl_block = new XTemplate('html/block.xtpl');
  198.     
  199.     $xtpl_block->reset('block');
  200.     $xtpl_block->assign('id', $block['id']);
  201.     $xtpl_block->assign('title', $block['title']);
  202.     $xtpl_block->assign('content', $block['content']);
  203.     
  204.     if (isset($block['links'])) {
  205.         $xtpl_block->assign('links', $block['links']);
  206.         $xtpl_block->parse('block.links');
  207.     }
  208.     
  209.     $xtpl_block->parse('block');
  210.     return $xtpl_block->text('block');
  211. }
