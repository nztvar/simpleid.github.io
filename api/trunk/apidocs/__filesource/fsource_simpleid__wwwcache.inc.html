  37. /**
  38.  *  Stores data into the cache.
  39.  *
  40.  * @param string $type the type of data in the cache
  41.  * @param string $key an identifier
  42.  * @param mixed $data the data to store
  43.  * @param int $time if present, sets the modification time of the cache file to this
  44.  *  time
  45.  */ 
  46. function cache_set($type, $key, $data, $time = NULL) {
  47.     $filename = _cache_get_name($type, $key);
  48.     $file = fopen($filename, 'w');
  49.     fwrite($file, serialize($data));
  50.     fclose($file);
  51.     
  52.     if ($time != NULL) {
  53.         touch($filename, $time);
  54.     }
  55. }

  57. /**
  58.  * Obtains data from the cache.
  59.  *
  60.  * @param string $type the type of data in the cache
  61.  * @param string $key an identifier
  62.  * @return mixed the data associated with the type and key, or NULL if the cache
  63.  *  does not contain the requested data.
  64.  */
  65. function cache_get($type, $key) {
  66.     $filename = _cache_get_name($type, $key);
  67.     
  68.     if (!file_exists($filename)) return NULL;
  69.     
  70.     return unserialize(file_get_contents($filename));
  71. }
  72.  
  73. /**
  74.  * Obtains all data of a particular type from the cache.
  75.  *
  76.  * @param string $type the type of data in the cache
  77.  * @return mixed an array of data associated with the type, or NULL if the cache
  78.  *  does not contain the requested data.
  79.  */
  80. function cache_get_all($type) {
  81.     $r = array();
  82.     
  83.     $dir = opendir(CACHE_DIR);
  84.     
  85.     while (($file = readdir($dir)) !== false) {
  86.         $filename = CACHE_DIR . '/' . $file;
  87.         
  88.         if ((filetype($filename) != "file") || (strpos($file, $type . '-') !== 0)) continue;
  89.         
  90.         $r[] = unserialize(file_get_contents($filename));
  91.     }
  92.     
  93.     closedir($dir);
  94.     
  95.     return $r;
  96. }

  98. /**
  99.  * Deletes data from the cache.
  100.  *
  101.  * @param string $type the type of data in the cache
  102.  * @param string $key an identifier
  103.  */
  104. function cache_delete($type, $key) {
  105.     $filename = _cache_get_name($type, $key);
  106.     if (file_exists($filename)) unlink($filename);
  107. }

  109. /**
  110.  * Garbage collects data stored the cache.  Data is deleted if it was stored
  111.  * for longer than the specified expiry
  112.  *
  113.  * @param int $expiry the expiry time, in seconds, after which data will be deleted
  114.  * @param string $type the type of data in the cache
  115.  */
  116. function cache_gc($expiry, $type = NULL) {
  117.     $dir = opendir(CACHE_DIR);
  118.     
  119.     while (($file = readdir($dir)) !== false) {
  120.         $filename = CACHE_DIR . '/' . $file;
  121.         
  122.         if (($type != NULL) && (strpos($file, $type . '-') !== 0)) continue;
  123.         
  124.         if ((filetype($filename) == "file") && (filectime($filename) < time() - $expiry)) {
  125.             unlink($filename);
  126.         }
  127.     }
  128.     
  129.     closedir($dir);
  130. }
  131.  
  132. /**
  133.  * Returns the time remaining, in seconds, before the data associated with the
  134.  * type and key become subject to garbage collection by {@link cache_gc()}.
  135.  *
  136.  * @param string $type the type of data in the cache
  137.  * @param string $key an identifier
  138.  * @param int $expiry the expiry time, in seconds, which would be passed onto the
  139.  *  {@link cache_gc()} function
  140.  * @return int the time remaining before expiry, rounded downwards,
  141.  *  or zero if the cache does not contain the requested data
  142.  * @since 0.8
  143.  */
  144. function cache_ttl($type, $key, $expiry) {
  145.     $filename = _cache_get_name($type, $key);
  146.     
  147.     if (!file_exists($filename)) return 0;
  148.     
  149.     return filectime($filename) - (time() - $expiry) - 1;
  150. }

  152. /**
  153.  * Returns the name of the cache data file, given a type and an identifier.
  154.  *
  155.  * @param string $type the type of data in the cache
  156.  * @param string $key an identifier
  157.  * @return string a file name
  158.  */
  159. function _cache_get_name($type, $key) {
  160.     return CACHE_DIR . '/' . $type . '-' . md5($key) . '.cache';
  161. }
