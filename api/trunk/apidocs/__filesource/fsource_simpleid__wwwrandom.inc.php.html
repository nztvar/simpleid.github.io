  40. /**
  41.  * Obtains a number of random bytes.  This function uses an entropy source specified
  42.  * in SIMPLEID_RAND_SOURCE.  If SIMPLEID_RAND_SOURCE is not available, the mt_rand()
  43.  * PHP function is used
  44.  *
  45.  * @param int $num_bytes the number of bytes to generate
  46.  * @return string a string containing random bytes
  47.  */
  48. function random_bytes($num_bytes) {
  49.     static $f = null;
  50.     $bytes = '';
  51.     if ($f === null) {
  52.         if (SIMPLEID_RAND_SOURCE === null) {
  53.             $f = FALSE;
  54.         } else {
  55.             $f = @fopen(SIMPLEID_RAND_SOURCE, "r");
  56.         }
  57.     }
  58.     if ($f === FALSE) {
  59.         $bytes = '';
  60.         for ($i = 0; $i < $num_bytes; $i += 4) {
  61.             $bytes .= pack('L', mt_rand());
  62.         }
  63.         $bytes = substr($bytes, 0, $num_bytes);
  64.     } else {
  65.         $bytes = fread($f, $num_bytes);
  66.     }
  67.     return $bytes;
  68. }

  70. /**
  71.  * Obtains a random string of a specified number of bytes of entropy.
  72.  *
  73.  * The function calls the {@link random_bytes()} function with the specified
  74.  * number of bytes, then converts to a string containing only alphanumeric
  75.  * characters (case sensitive), plus the characters ., _ and -.
  76.  *
  77.  * The conversion method is based on the Base64 encoding.  However, non-standard
  78.  * characters are used so that users are not confused and attempt to decode
  79.  * the returned string.
  80.  *
  81.  * @param int $num_bytes the approximate number of bytes of entropy in the
  82.  *  random string
  83.  * @return string the random string
  84.  */
  85. function random_secret($num_bytes = 32) {
  86.     return strtr(base64_encode(random_bytes($num_bytes)),
  87.         'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=',
  88.         '-_.9876543210zyxwvutsrqponmlkjihgfedcbaZYXWVUTSRQPONMLKJIHGFEDCBA');
  89. }

  91. /**
  92.  * Generates a relatively unique identifier which can be used as, among other things,
  93.  * an OpenID association handle or an OAuth client identifier.  The identifier
  94.  * returned is at least 24 characters long and contains only hexadecimal characters.
  95.  *
  96.  * Note that the identifier returned is not cryptographically secure.
  97.  *
  98.  * @return string a relatively unique identifier
  99.  */
  100. function random_id() {
  101.     $timeofday = gettimeofday();
  102.     return vsprintf('%08x%08x', $timeofday) . bin2hex(random_bytes(4));
