  62. function user_init($q = NULL) {
  63.     global $user;
  64.     global $xtpl;
  65.     
  66.     log_debug('user_init');
  67.     
  68.     $user = NULL;
  69.     
  70.     session_set_cookie_params(0);
  71.     session_name('SESS' . md5(SIMPLEID_BASE_URL));
  72.     session_start();
  73.     
  74.     if (isset($_SESSION['user']) && (cache_get('user', $_SESSION['user']) == session_id())) {
  75.         $user = user_load($_SESSION['user']);
  76.         
  77.         // If user has just been actively been authenticated in the previous request, then we
  78.         // make it as actively authenticated in this request.
  79.         if (isset($_SESSION['user_auth_active']) && $_SESSION['user_auth_active']) {
  80.             $user['auth_active'] = true;
  81.             unset($_SESSION['user_auth_active']);
  82.         }
  83.     } elseif (isset($_COOKIE[_user_autologin_cookie()])) {
  84.         if (($q == 'login') || ($q == 'logout')) return;
  85.         user_autologin_verify();
  86.     }
  87. }

  89. /**
  90.  * Loads user data for a specified user name.
  91.  *
  92.  * @param string $uid the name of the user to load
  93.  * @return mixed data for the specified user, or NULL if the user name does not
  94.  *  exist
  95.  * @see user_load_from_identity()
  96.  */
  97. function user_load($uid) {
  98.     if (store_user_exists($uid)) {
  99.         $user = store_user_load($uid);
  100.         $user["uid"] = $uid;
  101.         
  102.         if (isset($user["identity"])) {
  103.             $user["local_identity"] = true;
  104.         } else {
  105.             $user["identity"] = simpleid_url('user/' . rawurlencode($uid));
  106.             $user["local_identity"] = false;
  107.         }
  108.         
  109.         return $user;
  110.     } else {
  111.         return NULL;
  112.     }
  113. }

  140. /**
  141.  * Attempts to log in a user, using the user name and password specified in the
  142.  * HTTP request.
  143.  */
  144. function user_login() {
  145.     global $user, $GETPOST;
  146.     
  147.     // If the user is already logged in, return
  148.     if (isset($user["uid"])) openid_indirect_response(simpleid_url(), '');
  149.     
  150.     $destination = (isset($GETPOST['destination'])) ? $GETPOST['destination'] : '';
  151.     $state = (isset($GETPOST['s'])) ? $GETPOST['s'] : '';
  152.     $query = ($state) ? 's=' . rawurlencode($state) : '';
  153.     
  154.     if (isset($_POST['op']) && $_POST['op'] == 'Cancel') {
  155.         global $version;
  156.         
  157.         $request = unpickle($state);
  158.         $version = openid_get_version($request);
  159.         
  160.         if (isset($request['openid.return_to'])) {
  161.             $return_to = $request['openid.return_to'];
  162.             $response = simpleid_checkid_error(FALSE);
  163.             simpleid_assertion_response($response, $return_to);
  164.         } else {
  165.             indirect_fatal_error('Login cancelled without a proper OpenID request.');
  166.         }
  167.         return;
  168.     }
  169.     
  170.     // We allow legacy login if the connection is via HTTPS or if SIMPLEID_ALLOW_LEGACY_LOGIN is true
  171.     $allow_legacy_login = (is_https() || SIMPLEID_ALLOW_LEGACY_LOGIN);
  172.     
  173.     if (!isset($_POST['name'])) $_POST['name'] = '';
  174.     if (!isset($_POST['pass'])) $_POST['pass'] = '';
  175.     if (!isset($_POST['digest'])) $_POST['digest'] = '';
  176.     
  177.     if (
  178.         ($_POST['name'] == '')
  179.         || (
  180.             ($allow_legacy_login && ($_POST['pass'] == '') && ($_POST['digest'] == ''))
  181.             || (!$allow_legacy_login && ($_POST['digest'] == ''))
  182.         )
  183.     ) {
  184.         if (isset($_POST['destination'])) {
  185.             // User came from a log in form.
  186.             set_message('You need to supply the user name and the password in order to log in.');
  187.         }
  188.         if (isset($_POST['nonce'])) cache_delete('user-nonce', $_POST['nonce']);
  189.         user_login_form($destination, $state);
  190.         return;
  191.     }
  192.     
  193.     if (!isset($_POST['nonce'])) {
  194.         if (isset($_POST['destination'])) {
  195.             // User came from a log in form.
  196.             set_message('You seem to be attempting to log in from another web page.  You must use this page to log in.');
  197.         }
  198.         user_login_form($destination, $state);
  199.         return;
  200.     }
  201.     
  202.     $time = strtotime(substr($_POST['nonce'], 0, 20));
  203.     // Some old versions of PHP does not recognise the T in the ISO 8601 date.  We may need to convert the T to a space
  204.     if (($time == -1) || ($time === FALSE)) $time = strtotime(strtr(substr($_POST['nonce'], 0, 20), 'T', ' '));
  205.     
  206.     if (!cache_get('user-nonce', $_POST['nonce'])) {
  207.         log_warn('Login attempt: Nonce ' . $_POST['nonce'] . ' not issued or is being reused.');
  208.         set_message('SimpleID detected a potential security attack on your log in.  Please log in again.');
  209.         user_login_form($destination, $state);
  210.         return;
  211.     } elseif ($time < time() - SIMPLEID_LOGIN_NONCE_EXPIRES_IN) {
  212.         log_notice('Login attempt: Nonce ' . $_POST['nonce'] . ' expired.');
  213.         set_message('The log in page has expired.  Please log in again.');
  214.         user_login_form($destination, $state);
  215.         return;
  216.     } else {
  217.         cache_delete('user-nonce', $_POST['nonce']);
  218.     }
  219.     
  220.     if (store_user_verify_credentials($_POST['name'], $_POST, $allow_legacy_login) === false) {
  221.         set_message('The user name or password is not correct.');
  222.         user_login_form($destination, $state);
  223.         return;
  224.     }
  225.     
  226.     // Set the current authentication time
  227.     $test_user = user_load($_POST['name']);
  228.     $test_user['auth_time'] = time();
  229.     user_save($test_user);
  230.     
  231.     // Set user has been actively authenticated this and the next request only
  232.     $test_user['auth_active'] = true;
  233.     $_SESSION['user_auth_active'] = true;
  234.     
  235.     _user_login($test_user);
  236.     log_info('Login successful: ' . $test_user['uid'] . '['. gmstrftime('%Y-%m-%dT%H:%M:%SZ', $test_user['auth_time']) . ']');
  237.     
  238.     
  239.     if (isset($_POST['autologin']) && ($_POST['autologin'] == 1)) user_autologin_create();
  240.  
  241.     openid_indirect_response(simpleid_url($destination, $query), '');
  242. }

  297. /**
  298.  * Displays a user login form.
  299.  *
  300.  * @param string $destination the SimpleID location to which the user is directed
  301.  *  if login is successful
  302.  * @param string $state the current SimpleID state, if required by the location
  303.  */
  304. function user_login_form($destination = '', $state = NULL) {    
  305.     global $xtpl;
  306.  
  307.     if ($state) {
  308.         $xtpl->assign('state', htmlspecialchars($state, ENT_QUOTES, 'UTF-8'));
  309.         $xtpl->parse('main.login.state');
  310.     }
  311.     
  312.     cache_gc(SIMPLEID_LOGIN_NONCE_EXPIRES_IN, 'user-nonce');
  313.     $nonce = openid_nonce();
  314.     cache_set('user-nonce', $nonce, 1);
  315.     
  316.     $base_path = get_base_path();
  317.     
  318.     $xtpl->assign('javascript', '<script src="' . $base_path . 'html/md5.js" type="text/javascript"></script><script src="' . $base_path . 'html/user-login.js" type="text/javascript"></script>');
  319.     
  320.     $security_class = (SIMPLEID_ALLOW_AUTOCOMPLETE) ? 'allow-autocomplete ' : '';
  321.     
  322.     if (is_https()) {
  323.         $security_class .= 'secure';
  324.         $xtpl->assign('security_message', 'Secure login using <strong>HTTPS</strong>.');
  325.     } else {
  326.         if (!SIMPLEID_ALLOW_LEGACY_LOGIN) {
  327.             $security_class .= 'unsecure login-digest';
  328.             $xtpl->assign('security_message', 'Your SimpleID configuration does not allow you to log in unsecurely.  Please enable JavaScript and try again, or see <a href="http://simpleid.sourceforge.net/documentation/using-simpleid/logging-simpleid">the SimpleID documentation</a> for more details.');
  329.             $xtpl->assign('security_disabled', 'disabled="disabled"');
  330.         } else {
  331.             $security_class .= 'unsecure login-digest';
  332.             $xtpl->assign('security_message', '<strong>WARNING:</strong>  Your password will be sent to SimpleID as plain text.');
  333.         }
  334.     }
  335.     $xtpl->assign('security_class', $security_class);
  336.     
  337.     extension_invoke_all('user_login_form', $destination, $state);
  338.     
  339.     header('X-Frame-Options: DENY');
  340.  
  341.     $xtpl->assign('title', 'Log In');
  342.     $xtpl->assign('page_class', 'dialog-page');
  343.     $xtpl->assign('destination', htmlspecialchars($destination, ENT_QUOTES, 'UTF-8'));
  344.     $xtpl->assign('nonce', htmlspecialchars($nonce, ENT_QUOTES, 'UTF-8'));
  345.     
  346.     $xtpl->parse('main.login');
  347.     $xtpl->parse('main.framekiller');
  348.     $xtpl->parse('main');
  349.     $xtpl->out('main');
  350. }

  352. /**
  353.  * Returns the user's public page.
  354.  * 
  355.  * @param string $uid the user ID
  356.  */
  357. function user_public_page($uid = NULL) {
  358.     global $xtpl, $user;
  359.     
  360.     $xtpl->assign('title', 'User Page');
  361.     if ($uid == NULL) {
  362.         header_response_code('400 Bad Request');
  363.         set_message('No user specified.');
  364.     } else {
  365.         $user = user_load($uid);
  366.         
  367.         if ($user == NULL) {
  368.             header_response_code('404 Not Found');
  369.             set_message('User <strong>' . htmlspecialchars($uid, ENT_QUOTES, 'UTF-8') . '</strong> not found.');
  370.         } else {
  371.             header('Vary: Accept');
  372.             
  373.             $content_type = negotiate_content_type(array('text/html', 'application/xml', 'application/xhtml+xml', 'application/xrds+xml'));
  374.             
  375.             if ($content_type == 'application/xrds+xml') {
  376.                 user_xrds($uid);
  377.                 return;
  378.             } else {
  379.                 header('X-XRDS-Location: ' . simpleid_url('xrds/' . rawurlencode($uid)));
  380.                 
  381.                 set_message('This is the user <strong>' . htmlspecialchars($uid, ENT_QUOTES, 'UTF-8') . '</strong>\'s SimpleID page.  It contains hidden information for the use by OpenID consumers.');
  382.                 
  383.                 $xtpl->assign('title', htmlspecialchars($uid, ENT_QUOTES, 'UTF-8'));
  384.                 $xtpl->assign('provider', htmlspecialchars(simpleid_url(), ENT_QUOTES, 'UTF-8'));
  385.                 $xtpl->assign('xrds', htmlspecialchars(simpleid_url('xrds/' . rawurlencode($uid)), ENT_QUOTES, 'UTF-8'));
  386.                 if ($user["local_identity"]) {
  387.                     $xtpl->assign('local_id', htmlspecialchars($user["identity"], ENT_QUOTES, 'UTF-8'));
  388.                 }
  389.             }
  390.         }
  391.     }
  392.     
  393.     $xtpl->parse('main.provider');
  394.     if ($user["local_identity"]) $xtpl->parse('main.local_id');
  395.     $xtpl->parse('main');
  396.     $xtpl->out('main');
  397. }

  417.         return;

  419.         header('X-XRDS-Location: ' . simpleid_url('ppid/' . rawurlencode($ppid), 'format=xrds'));
  420.                 
  421.         $xtpl->assign('title', 'Private Personal Identifier');

  508.  * @param string $id the ID of the series of auto login cookies,  Cookies
  509.  *  belonging to the same user and computer have the same ID.  If none is specified,
  510.  *  one will be generated
  511.  * @param int $expires the time at which the cookie will expire.  If none is specified
  512.  *  the time specified in {@link SIMPLEID_USER_AUTOLOGIN_EXPIRES_IN} will be
  513.  *  used
  514.  *
  515.  */
  516. function user_autologin_create($id = NULL, $expires = NULL) {
  517.     global $user;
  518.     
  519.     if ($expires == NULL) {
  520.         log_debug('Automatic login token created for ' . $user['uid']);
  521.     } else {
  522.         log_debug('Automatic login token renewed for ' . $user['uid']);
  523.     }
  524.     
  525.     if ($id == NULL) $id = get_form_token(mt_rand());
  526.     if ($expires == NULL) $expires = time() + SIMPLEID_USER_AUTOLOGIN_EXPIRES_IN;
  527.     $token = get_form_token(mt_rand());
  528.     
  529.     cache_set('autologin-'. md5($user['uid']), $id, array('token' => $token, 'expires' => $expires, 'ip' => $_SERVER['REMOTE_ADDR']));
  530.     
  531.     setcookie(_user_autologin_cookie(), $user['uid'] . ':' . $id . ':' . $token, $expires);
  532. }

  534. /**
  535.  * Verifies a auto login cookie.  If valid, log in the user automatically.
  536.  */
  537. function user_autologin_verify() {
  538.     $cookie = $_COOKIE[_user_autologin_cookie()];
  539.     
  540.     list($uid, $id, $token) = explode(':', $cookie);
  541.     log_debug('Automatic login token detected for ' . $uid);
  542.     
  543.     cache_gc(SIMPLEID_USER_AUTOLOGIN_EXPIRES_IN, 'autologin-' . md5($uid));
  544.     $cache = cache_get('autologin-' . md5($uid), $id);
  545.     
  546.     if (!$cache) {  // Cookie doesn't exist
  547.         log_notice('Automatic login: Token does not exist on server');
  548.         return;
  549.     }
  550.     
  551.     if ($cache['expires'] < time()) {  // Cookie expired
  552.         log_notice('Automatic login: Token on server expired');
  553.         return;
  554.     }
  555.     
  556.     if ($cache['token'] != $token) {
  557.         log_warn('Automatic login: Token on server does not match');
  558.         // Token not the same - panic
  559.         cache_gc(0, 'autologin-' . md5($uid));
  560.         user_autologin_invalidate();
  561.         return;
  562.     }
  563.     
  564.     // Load the user, tag it as an auto log in
  565.     $user = user_load($uid);
  566.     
  567.     if ($user != NULL) {
  568.         log_debug('Automatic login token accepted for ' . $uid);
  569.         
  570.         $user['autologin'] = TRUE;
  571.         _user_login($user);
  572.     
  573.         // Renew the token
  574.         user_autologin_create($id, $cache['expires']);
  575.     } else {
  576.         log_warn('Automatic login token accepted for ' . $uid . ', but no such user exists');
  577.     }
  578. }
