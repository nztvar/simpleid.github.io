  24. /**

  54. /**
  55.  * Creates a bignum.
  56.  *
  57.  * @param mixed $str An integer, a string in base 10, or a byte stream in base 256
  58.  * @param int $base either 10 or 256
  59.  * @return resource a bignum
  60.  */
  61. function bignum_new($str, $base = 10) {
  62.     if ($base == 10) {
  63.         if (BIGNUM_GMP) {
  64.             return gmp_init($str, 10);
  65.         } else {
  66.             return $str;
  67.         }
  68.     } elseif ($base == 256) {
  69.         $bytes = array_merge(unpack('C*', $str));
  70.  
  71.         $num = bignum_new(0);
  72.   
  73.         foreach ($bytes as $byte) {
  74.             $num = bignum_mul($num, 256);
  75.             $num = bignum_add($num, bignum_new($byte));
  76.         }
  77.         return $num;
  78.     }
  79.     return FALSE;
  80. }

  82. /**
  83.  * Converts a bignum into a string representation (base 10) or a byte stream
  84.  * (base 256)
  85.  *
  86.  * @param resource $num the bignum
  87.  * @param int $base either 10 or 256
  88.  * @return string the converted bignum
  89.  */
  90. function bignum_val($num, $base = 10) {
  91.     if (BIGNUM_GMP) {
  92.         $base10 = gmp_strval($num, 10);
  93.     } else {
  94.         $base10 = $num;
  95.     }
  96.  
  97.     if ($base == 10) return $base10;
  98.     if ($base == 256) {
  99.         $cmp = bignum_cmp($num, 0);
  100.         if ($cmp < 0) {
  101.             return FALSE;
  102.         }
  103.     
  104.         if ($cmp == 0) {
  105.             return "\x00";
  106.         }
  107.     
  108.         $bytes = array();
  109.       
  110.         while (bignum_cmp($num, 0) > 0) {
  111.             array_unshift($bytes, bignum_mod($num, 256));
  112.             $num = bignum_div($num, 256);
  113.         }
  114.       
  115.         if ($bytes && ($bytes[0] > 127)) {
  116.             array_unshift($bytes, 0);
  117.         }
  118.       
  119.         $byte_stream = '';
  120.         foreach ($bytes as $byte) {
  121.             $byte_stream .= pack('C', $byte);
  122.         }
  123.       
  124.         return $byte_stream;
  125.     }
  126.     return FALSE;
  127. }

  205. /**
  206.  * Raise a number into power with modulo
  207.  *
  208.  * @param resource $base the base
  209.  * @param resource $exp the exponent
  210.  * @param resource $mod the modulo
  211.  * @return resource a bignum representing base ^ exp mod mod
  212.  */
  213. function bignum_powmod($base, $exp, $mod) {
  214.     if (BIGNUM_GMP) {
  215.         return gmp_powm($base, $exp, $mod);
  216.     } elseif (function_exists('bcpowmod')) {
  217.         return bcpowmod($base, $exp, $mod);
  218.     } else {
  219.         $square = bignum_mod($base, $mod);
  220.         $result = 1;
  221.         while (bignum_cmp($exp, 0) > 0) {
  222.             if (bignum_mod($exp, 2)) {
  223.                 $result = bignum_mod(bignum_mul($result, $square), $mod);
  224.             }
  225.             $square = bignum_mod(bignum_mul($square, $square), $mod);
  226.             $exp = bignum_div($exp, 2);
  227.         }
  228.         return $result;
  229.     }
  230. }
