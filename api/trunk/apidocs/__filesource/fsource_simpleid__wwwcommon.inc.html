  82. /**
  83.  * Determines whether the current connection with the user agent is via

  94. function is_https() {
  95.     return (isset($_SERVER['HTTPS']) && ($_SERVER['HTTPS'] == 'on'))
  96.         || (isset($_SERVER['HTTP_X_FORWARDED_PROTO']) && (strtolower($_SERVER['HTTP_X_FORWARDED_PROTO']) == 'https'))
  97.         || (isset($_SERVER['HTTP_FRONT_END_HTTPS']) && ($_SERVER['HTTP_FRONT_END_HTTPS'] == 'on'));

  100. /**
  101.  * Content type negotiation using the Accept Header.
  102.  *
  103.  * Under HTTP, the user agent is able to negoatiate the content type returned with
  104.  * the server using HTTP Accept header.  This header contains a comma-delimited
  105.  * list of items (e.g. content types) which the user agent is able to
  106.  * accept, ranked by a quality parameter.
  107.  *
  108.  * This function takes the header from the user agent, compares it against the
  109.  * content types which the server can provide, then returns the item which the highest
  110.  * quality which the server can provide.
  111.  *
  112.  * @param array $content_types an array of content types which the server can
  113.  *  provide
  114.  * @param string $accept_header the header string provided by the user agent.
  115.  *  If NULL, this defaults to $_SERVER['HTTP_ACCEPT'] if available
  116.  * @return string the negotiated content type, FALSE if $accept_header is NULL and
  117.  *  the user agent did not provide an Accept header, or NULL if the negotiation is
  118.  *  unsuccessful
  119.  *
  120.  * @since 0.8
  121.  *
  122.  */
  123. function negotiate_content_type($content_types, $accept_header = NULL) {
  124.     $content_types = array_map("strtolower", $content_types);
  125.     if (($accept_header == NULL) && isset($_SERVER['HTTP_ACCEPT'])) $accept_header = $_SERVER['HTTP_ACCEPT'];
  126.     
  127.     if ($accept_header) {
  128.         $acceptible = preg_split('/\s*,\s*/', strtolower(trim($accept_header)));
  129.         for ($i = 0; $i < count($acceptible); $i++) {
  130.             $split = preg_split('/\s*;\s*q\s*=\s*/', $acceptible[$i], 2);
  131.             $item = strtolower($split[0]);
  132.             
  133.             if (count($split) == 1) {
  134.                 $q = 1.0;
  135.             } else {
  136.                 $q = doubleval($split[1]);
  137.             }
  138.             
  139.             if ($q > 0.0) {
  140.                 if (in_array($item, $content_types)) {
  141.                     if ($q == 1.0) {
  142.                         return $item;
  143.                     }
  144.                     $candidates[$item] = $q;
  145.                 } else {
  146.                     $item = preg_quote($item, '/');
  147.                     $item = strtr($item, array('\*' => '[^\\x00-\\x20\\x22\\x28\\x29\\x2c\\x2e\\x3a-\\x3c\\x3e\\x40\\x5b-\\x5d\\x7f-\\xff]+'));
  148.                     
  149.                     foreach ($content_types as $value) {
  150.                         if (preg_match("/^$item$/", $value)) {
  151.                             if ($q == 1.0) {
  152.                                 return $value;
  153.                             }
  154.                             $candidates[$value] = $q;
  155.                             break;
  156.                         }
  157.                     }
  158.                 }
  159.             }
  160.         }
  161.         if (isset($candidates)) {
  162.             arsort($candidates);
  163.             reset($candidates);
  164.             return key($candidates);
  165.         }
  166.         return NULL;
  167.     } else {
  168.         // No headers
  169.         return FALSE;
  170.     }
  171. }

  224. function get_base_path() {

  227.     if (!$base_path) {
  228.         if ((substr(SIMPLEID_BASE_URL, -1) == '/') || (substr(SIMPLEID_BASE_URL, -9) == 'index.php')) {
  229.             $url = SIMPLEID_BASE_URL;

  241. /**
  242.  * Obtains a SimpleID URL.  URLs produced by SimpleID should use this function.
  243.  *
  244.  * @param string $q the q parameter
  245.  * @param string $params a properly encoded query string
  246.  * @param bool $relative whether a relative URL should be returns
  247.  * @return string the url
  248.  *
  249.  * @since 0.7
  250.  */
  251. function simpleid_url($q = '', $params = '', $relative = false) {
  252.     if ($relative) {
  253.         $url = get_base_path();
  254.     } else {
  255.         // Make sure that the base has a trailing slash
  256.         if ((substr(SIMPLEID_BASE_URL, -1) == '/') || (substr(SIMPLEID_BASE_URL, -9) == 'index.php')) {
  257.             $url = SIMPLEID_BASE_URL;
  258.         } else {
  259.             $url = SIMPLEID_BASE_URL . '/';
  260.         }
  261.     }
  262.     if (SIMPLEID_CLEAN_URL) {
  263.         $url .= $q . (($params == '') ? '' : '?' . $params);
  264.     } elseif (($q == '') && ($params == '')) {
  265.         $url .= '';
  266.     } else {
  267.         $url .= 'index.php?q=' . $q . (($params == '') ? '' : '&' . $params);
  268.     }
  269.     return $url;
  270. }

  272. /**
  273.  * Obtains a form token given a form ID.
  274.  *
  275.  * Form tokens are used in SimpleID forms to guard against cross-site forgery
  276.  * attacks.
  277.  *
  278.  * @param string $id the form ID
  279.  * @return string a form token
  280.  */
  281. function get_form_token($id) {
  282.     global $user;
  283.  
  284.     if (store_get('site-token') == NULL) {
  285.         $site_token = mt_rand();
  286.         store_set('site-token', $site_token);
  287.     } else {
  288.         $site_token = store_get('site-token');
  289.     }
  290.     
  291.     if ($user == NULL) {
  292.         return md5($id . $site_token);
  293.     } else {
  294.         return md5(session_id() . $id . $site_token);
  295.     }
  296. }

  343. /**
  344.  * Invokes a hook in all the loaded extensions.
  345.  *
  346.  * @param string $function the name of the hook to call
  347.  * @param mixed $args the arguments to the hook
  348.  * @return array the return values from the hook
  349.  */
  350. function extension_invoke_all() {
  351.     global $simpleid_extensions;
  352.     
  353.     $args = func_get_args();
  354.     $function = array_shift($args);
  355.     $return = array();
  356.     
  357.     foreach ($simpleid_extensions as $extension) {
  358.         if (function_exists($extension . '_' . $function)) {
  359.             log_debug('extension_invoke_all: ' . $extension . '_' . $function);
  360.             $result = call_user_func_array($extension . '_' . $function, $args);
  361.             if (isset($result) && is_array($result)) {
  362.                 $return = array_merge($return, $result);
  363.             } elseif (isset($result)) {
  364.                 $return[] = $result;
  365.             } 
  366.         }
  367.     }
  368.     
  369.     return $return;
  370. }
